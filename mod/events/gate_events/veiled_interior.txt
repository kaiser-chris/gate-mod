namespace = gate_veiled_interior

gate_veiled_interior.1 = {
	type = country_event
	dlc = dlc_magic_gate
	placement = capital

	title = gate_veiled_interior.1.t
	desc = gate_veiled_interior.1.d
	flavor = gate_veiled_interior.1.f

	event_image = {
		video = "unspecific_signed_contract"
	}

	gui_window = event_window_widescreen_classic

	trigger = {
		any_scope_state = {
			OR = {
				state_region = s:STATE_EASTERN_NEW_GUINEA
				state_region = s:STATE_WESTERN_NEW_GUINEA
			}
		}
	}

	is_popup = yes

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = gfx/interface/icons/event_icons/event_map.dds

	duration = 3

	option = {
		name = gate_veiled_interior.1.a
		default_option = yes
		highlighted_option = yes
		custom_tooltip = {
			text = gate_veiled_interior.1.a.tooltip
			every_state = {
				limit = {
					OR = {
						state_region = s:STATE_EASTERN_NEW_GUINEA
						state_region = s:STATE_WESTERN_NEW_GUINEA
					}
					is_largest_state_in_region = yes
				}
				create_pop = {
					culture = melanesian
					religion = animist
					size = 500000
				}
			}
		}
		s:STATE_EASTERN_NEW_GUINEA = {
			add_arable_land = { 20 30 }
		}
		s:STATE_WESTERN_NEW_GUINEA = {
			add_arable_land = { 20 30 }
		}
		every_state = {
			limit = {
				OR = {
					state_region = s:STATE_EASTERN_NEW_GUINEA
					state_region = s:STATE_WESTERN_NEW_GUINEA
				}
				is_largest_state_in_region = yes
			}
			add_modifier = {
				name = modifier_veiled_interior_lifted
				days = long_modifier_time
				is_decaying = yes
			}
		}
		add_modifier = {
			name = unification_prestige
			days = very_long_modifier_time
			is_decaying = yes
		}
	}
}

gate_veiled_interior.2 = {
	type = country_event
	dlc = dlc_magic_gate
	placement = scope:expedition_state

	title = gate_veiled_interior.2.t
	desc = gate_veiled_interior.2.d
	flavor = gate_veiled_interior.2.f

	event_image = {
		video = "middleeast_jungle_expedition"
	}

	trigger = {
		any_scope_state = {
			OR = {
				state_region = s:STATE_EASTERN_NEW_GUINEA
				state_region = s:STATE_WESTERN_NEW_GUINEA
			}
		}
	}

	immediate = {
		random_scope_state = {
			limit = {
				OR = {
					state_region = s:STATE_EASTERN_NEW_GUINEA
					state_region = s:STATE_WESTERN_NEW_GUINEA
				}
			}
			save_scope_as = expedition_state
		}
		if = {
			limit = {
				NOT = {
					exists = scope:journal_entry
				}
			}
			je:je_gate_veiled_interior ?= {
				save_scope_as = journal_entry
			}
		}
		scope:journal_entry = {
			set_variable = {
				name = gate_veiled_interior_phase
				value = flag:je_gate_veiled_interior_phase_early_contact
			}
		}
	}

	is_popup = yes

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = gfx/interface/icons/event_icons/event_map.dds

	option = {
		name = gate_veiled_interior.2.a
		default_option = yes
		custom_tooltip = je_gate_veiled_interior_effect_phase_change_early_contact
		add_modifier = {
			name = modifier_gate_veiled_interior_social_harmony
			years = 3
		}
	}
}
