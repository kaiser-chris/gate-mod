namespace = monster_attack

monster_attack.1 = {
    type = country_event
    dlc = dlc_magic_gate
    placement = scope:monster_horde_state

    title = monster_attack.1.t
    desc = monster_attack.1.d
    flavor = monster_attack.1.f

    event_image = {
        video = "southamerica_war_civilians"
    }

    on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

    icon = "gfx/interface/icons/event_icons/event_skull.dds"

    duration = 3

    trigger = {
        is_player = yes
    }

    immediate = {
        if = {
            limit = {
                NOT = {
                    has_global_variable = monster_horde_happened
                }
            }
            set_global_variable = monster_horde_happened
            every_country = {
                limit = {
                    is_gate_planet_country = no
                }
                add_journal_entry = {
                    type = je_monster_congress
                }
            }
        }
        random_state = {
            limit = {
                state_region = {
                    var:mana_density >= medium_mana_density_threshhold
                }
            }
            save_scope_as = monster_horde_state
            owner = {
                save_scope_as = monster_horde_country
            }
        }
        # For five to ten years after the horde
        # no more hordes will happen
        set_global_variable = {
            name = monster_horde_grace_period
            years = { 5 10 }
        }
        scope:monster_horde_state = {
            state_region = {
                start_harvest_condition = monster_horde
            }
        }
    }

    # Our countries state
    option = {
        name = monster_attack.1.a1
        add_loyalists = {
            value = very_large_radicals
        }
        add_modifier = {
            name = monster_horde_cost_modifier
            months = 12
            multiplier = money_amount_multiplier_very_small
        }
        trigger = {
            owner = scope:monster_horde_country
        }
    }
    option = {
        name = monster_attack.1.b1
        add_loyalists = {
            value = very_large_radicals
        }
        add_modifier = {
            name = monster_horde_cost_modifier
            months = 12
            multiplier = money_amount_multiplier_small
        }
        hidden_effect = {
            trigger_event = {
                id = researching_magic.1
                months = 3
            }
        }
        make_magic_researcher_busy = yes
        trigger = {
            has_variable = magic_researcher
            var:magic_researcher ?= {
                is_busy = no
            }
            owner = scope:monster_horde_country
        }
    }
    option = {
        name = monster_attack.1.c1
        default_option = yes
        add_radicals = {
            value = very_large_radicals
        }
        trigger = {
            owner = scope:monster_horde_country
        }
    }

    # Other countries state
    option = {
        name = monster_attack.1.a2
        change_relations = {
            country = scope:monster_horde_country
            value = 20
        }
        add_modifier = {
            name = monster_horde_cost_modifier
            months = 12
            multiplier = money_amount_multiplier_very_small
        }
        trigger = {
            NOT = {
                owner = scope:monster_horde_country
            }
        }
    }
    option = {
        name = monster_attack.1.b2
        change_relations = {
            country = scope:monster_horde_country
            value = 10
        }
        add_modifier = {
            name = monster_horde_cost_modifier
            months = 12
            multiplier = money_amount_multiplier_small
        }
        hidden_effect = {
            trigger_event = {
                id = researching_magic.1
                months = 3
            }
        }
        make_magic_researcher_busy = yes
        trigger = {
            has_variable = magic_researcher
            var:magic_researcher ?= {
                is_busy = no
            }
            NOT = {
                owner = scope:monster_horde_country
            }
        }
    }
    option = {
        name = monster_attack.1.c2
        default_option = yes
        change_relations = {
            country = scope:monster_horde_country
            value = -30
        }
        change_infamy = 5
        trigger = {
            NOT = {
                owner = scope:monster_horde_country
            }
        }
    }
}
