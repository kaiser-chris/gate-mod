namespace = gate_arcane_council

gate_arcane_council.1 = {
	type = country_event
	dlc = dlc_magic_gate
	placement = scope:target_state

	title = gate_arcane_council.1.t
	desc = gate_arcane_council.1.d
	flavor = gate_arcane_council.1.f

	event_image = {
		video = "unspecific_devastation"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_skull.dds"

	duration = 3

	trigger = {
		gate_trigger_incident_events = yes
		gate_is_arcane_council_member = no
		NOT = {
			has_global_variable = gate_spellcaster_incidents_grace_period
		}
	}

	immediate = {
		random_country = {
			limit = {
				is_gate_magic_country = yes
				NOT = {
					this = root
				}
				has_overlapping_interests = root
			}
			save_scope_as = source_country
		}
		random_scope_state = {
			save_scope_as = target_state
		}
		set_variable = gate_foreign_mage_incident
		set_variable = {
			name = gate_foreign_mage_incident_cooldown
			months = { 24 36 }
		}
	}

	option = {
		name = gate_arcane_council.1.a
		default_option = yes
		change_relations = {
			country = scope:source_country
			value = -20
		}
		scope:target_state ?= {
			add_loyalists_in_state = {
				value = 0.02
			}
			add_devastation = 10
		}
	}

	option = {
		name = gate_arcane_council.1.b
		scope:target_state ?= {
			add_radicals_in_state = {
				value = 0.05
			}
			add_devastation = 10
		}
	}

	option = {
		name = gate_arcane_council.1.c
		scope:target_state ?= {
			add_radicals_in_state = {
				value = 0.02
			}
			add_devastation = 5
		}
		add_modifier = {
			name = modifier_gate_arcane_council_incident_cost
			multiplier = money_amount_multiplier_medium
			months = 12
		}
	}
}

gate_arcane_council.2 = {
	type = country_event
	dlc = dlc_magic_gate
	placement = scope:target_state

	title = gate_arcane_council.2.t
	desc = gate_arcane_council.2.d
	flavor = gate_arcane_council.2.f

	event_image = {
		video = "asia_dead_cattle_poor_harvest"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_skull.dds"

	duration = 3

	trigger = {
		gate_trigger_incident_events = yes
		gate_is_arcane_council_member = no
		any_scope_state = {
			state_region = {
				NOT = {
					has_harvest_condition = drought
				}
			}
		}
		NOT = {
			has_global_variable = gate_spellcaster_incidents_grace_period
		}
	}

	immediate = {
		random_country = {
			limit = {
				is_gate_magic_country = yes
				NOT = {
					this = root
				}
				has_overlapping_interests = root
			}
			save_scope_as = source_country
		}
		random_scope_state = {
			limit = {
				state_region = {
					NOT = {
						has_harvest_condition = drought
					}
				}
			}
			save_scope_as = target_state
		}
		set_variable = gate_foreign_mage_incident
		set_variable = {
			name = gate_foreign_mage_incident_cooldown
			months = { 24 36 }
		}
	}

	option = {
		name = gate_arcane_council.1.a
		default_option = yes
		change_relations = {
			country = scope:source_country
			value = -20
		}
		scope:target_state ?= {
			add_loyalists_in_state = {
				pop_type = farmers
				value = 0.05
			}
			state_region = {
				start_harvest_condition_with_params = {
					type = drought
					duration = 12
					intensity = 3
				}
			}
		}
	}

	option = {
		name = gate_arcane_council.1.b
		scope:target_state ?= {
			add_radicals_in_state = {
				value = 0.10
				pop_type = farmers
			}
			state_region = {
				start_harvest_condition_with_params = {
					type = drought
					duration = 12
					intensity = 3
				}
			}
		}
	}

	option = {
		name = gate_arcane_council.1.c
		scope:target_state ?= {
			add_radicals_in_state = {
				value = 0.02
				pop_type = farmers
			}
			state_region = {
				start_harvest_condition_with_params = {
					type = drought
					duration = 12
					intensity = 3
				}
			}
		}
		add_modifier = {
			name = modifier_gate_arcane_council_incident_cost
			multiplier = money_amount_multiplier_medium
			months = 12
		}
	}
}
