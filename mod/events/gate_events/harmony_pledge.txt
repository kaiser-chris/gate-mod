gate_harmony_pledge.1 = {
	type = country_event
	dlc = dlc_magic_gate
	placement = scope:event_state

	title = gate_harmony_pledge.1.t
	desc = gate_harmony_pledge.1.d
	flavor = gate_harmony_pledge.1.f

	event_image = {
		video = "unspecific_ruler_speaking_to_people"
	}

	trigger = {
		any_scope_pop = {
			culture = cu:gnomish
		}
	}

	immediate = {
		random_scope_state = {
			limit = {
				any_scope_pop = {
					culture = cu:gnomish
				}
			}
			save_scope_as = event_state
		}
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	cooldown = {
		months = { 12 24 }
	}

	icon = gfx/interface/icons/religion_icons/gate_harmony_pledge.dds

	duration = 3

	option = {
		name = gate_harmony_pledge.1.a
		default_option = yes
	}

	option = {
		name = gate_harmony_pledge.1.b
		default_option = yes
	}

	option = {
		name = gate_harmony_pledge.1.c
		default_option = yes
	}
}
