gate_harmony_pledge.1 = {
	type = country_event
	dlc = dlc_magic_gate
	placement = scope:event_state

	title = gate_harmony_pledge.1.t
	desc = gate_harmony_pledge.1.d
	flavor = gate_harmony_pledge.1.f

	event_image = {
		video = "ip3_official_reading_dictate"
	}

	trigger = {
		any_scope_pop = {
			culture = cu:gnomish
		}
	}

	immediate = {
		random_scope_state = {
			limit = {
				any_scope_pop = {
					culture = cu:gnomish
				}
			}
			save_scope_as = event_state
		}
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	cooldown = {
		months = { 12 24 }
	}

	icon = gfx/interface/icons/religion_icons/gate_harmony_pledge.dds

	duration = 3

	#TODO: MasterOfGrey balance
	option = {
		name = gate_harmony_pledge.1.a
		add_gate_society_harmony_progress = {
			amount = 5
		}
		je:je_gate_harmony_pledge ?= {
			add_modifier = {
				name = modifier_gate_harmony_pledge_enthusiasm_without_boundaries
				months = 12
			}
		}
	}

	#TODO: MasterOfGrey balance
	option = {
		name = gate_harmony_pledge.1.b
		default_option = yes
		add_gate_society_harmony_progress = {
			amount = -5
		}
		scope:event_state ?= {
			add_radicals_in_state = {
				culture = cu:gnomish
				value = 0.02
			}
		}
	}
}

	option = {

			base = 10
			modifier = {
				trigger = {
					bureaucracy < 0		# Modifier reduces bureaucracy.
				}
				divide = 2				# Divide proportionally if misaligned.
			}
			modifier = {
				trigger = {
					"je:je_gate_harmony_pledge.scripted_bar_progress(pb_gate_harmony_pledge_economy)" < 55
				}
				add = -5	# Don't flip across the centreline if there's a better option. 
			}

				trigger = {
					"je:je_gate_harmony_pledge.scripted_bar_progress(pb_gate_harmony_pledge_economy)" < 45

				value = 0	# If current value is < 40 + add_gate_economy_harmony_progress do not move further from harmony.
			}
		}
		add_gate_economy_harmony_progress = {
			amount = -5
		}
		je:je_gate_harmony_pledge ?= {
			add_modifier = {
				name = modifier_gate_harmony_pledge_rewritten_ledger
				months = 12
				multiplier = -1
			}
		}
	}

	option = {
		name = gate_harmony_pledge.2.b
		ai_chance = {
			base = 10
			modifier = {
				trigger = {
					bureaucracy < 0		# Modifier adds bureaucracy.
				}
				multiply = 1			# Use Modifier Multiplier if aligned.
			}
			modifier = {
				trigger = {
					"je:je_gate_harmony_pledge.scripted_bar_progress(pb_gate_harmony_pledge_economy)" < 48
				}
				add = -5	# Don't flip across the centreline if there's a better option. 
			}
			modifier = {
				trigger = {
					"je:je_gate_harmony_pledge.scripted_bar_progress(pb_gate_harmony_pledge_economy)" > 58
				}
				value = 0	# If current value is < 60 - add_gate_economy_harmony_progress do not move further from harmony.
			}
		}
		default_option = yes
		add_gate_economy_harmony_progress = {
			amount = 2
		}
		je:je_gate_harmony_pledge ?= {
			add_modifier = {
				name = modifier_gate_harmony_pledge_rewritten_ledger
				months = 12
			}
		}
	}

	option = {
		name = gate_harmony_pledge.2.c
		ai_chance = {
			base = 10
			modifier = {
				trigger = {
					bureaucracy < 0		# Modifier adds bureaucracy.
				}
				multiply = 2			# Use Modifier Multiplier if aligned.
			}
			modifier = {
				trigger = {
					"je:je_gate_harmony_pledge.scripted_bar_progress(pb_gate_harmony_pledge_<type>)" > 45
				}
				add = -5	# If current value is > 50 - add_gate_<type>_harmony_progress don't flip across the centreline if there's a better option. 
			}
			modifier = {
				trigger = {
					"je:je_gate_harmony_pledge.scripted_bar_progress(pb_gate_harmony_pledge_<type>)" > 55
				}
				value = 0	# If current value is > 60 - add_gate_<type>_harmony_progress do not move further from harmony.
