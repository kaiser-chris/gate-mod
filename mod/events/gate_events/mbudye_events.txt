	dlc = dlc_magic_gate
	placement = scope:mbudye_event_state

	immediate = {
		scope_random_lunda_subject = yes
	}

	trigger = {
		# Only one crime at a time
		NOT = {
			has_variable = mbudye_investigation
		}
	}

	title = mbudye.10.t
	desc = mbudye.10.desc
	flavor = mbudye.10.f

	event_image = {
		video = "africa_public_protest"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/religion_icons/golden_faith.dds"

	duration = 3

	option = {
		name = mbudye.10.a
		default_option = yes
		add_radicals = {
			religion = rel:animist
			value = large_radicals
		}
		custom_tooltip = {
			text = je_gate_mbudye_change_fear
		}
		add_conversion_progress = {
			country = scope:mbudye_event_country
			conversion_progress = 10
			support_change = -5
		}
		ai_chance = {
			base = 10
			modifier = {
				trigger = {
					check_mbudye_support = {
						support_change = -5
					}
				}
				add = 90
			}
		}
	}

	option = {
		name = mbudye.10.b
		add_conversion_progress = {
			country = scope:mbudye_event_country
			conversion_progress = 0
			support_change = 2
		}
		hidden_effect = {
			trigger_random_mbudye_investigation_result = {
				chance_success = 90
				investigation_duration = 3
			}
			set_variable = {
				name = mbudye_investigation
				value = flag:mbudye_defaced_temple_flag
			}
		}
		ai_chance = {
			base = 10
			modifier = {
				trigger = {
					check_mbudye_support = {
						support_change = 2
					}
				}
				add = 90
			}
		}
	}

	option = {
		name = mbudye.10.c
		add_radicals = {
			religion = rel:golden_religion
			value = small_radicals
		}
		add_conversion_progress = {
			country = scope:mbudye_event_country
			conversion_progress = 1
			support_change = 4
		}
		ai_chance = {
			base = 10
			modifier = {
				trigger = {
					check_mbudye_support = {
						support_change = 4
					}
				}
				add = 90
			}
		}
	}
}
