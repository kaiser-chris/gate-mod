gate_mana_saturation.2 = {
	type = country_event
	dlc = dlc_magic_gate
	placement = scope:ritual_state
	category = gate_gate_mana_saturation

	title = gate_mana_saturation.2.t
	desc = gate_mana_saturation.2.d
	flavor = gate_mana_saturation.2.f

	event_image = {
		video = "southamerica_slave_chains"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_skull.dds"

	duration = 3

	trigger = {
		num_pending_events:gate_gate_mana_saturation = 0
	}

	immediate = {
		random_scope_state = {
			limit = {
				state_region = {
					var:mana_density < 100
				}
			}
			save_scope_as = ritual_state
			state_region = {
				save_scope_as = ritual_state_region
			}
		}
	}

	option = {
		name = gate_mana_saturation.2.a
		change_infamy = 10
		kill_population = {
			value = 10000
			pop_type = slaves
		}
		custom_tooltip = {
			text = gate_mana_saturation.2.a_progress_tt
			scope:ritual_state_region = {
				change_variable = {
					name = mana_density
					add = 50
					max = 100
				}
			}
		}
		set_variable = {
			name = slave_ritual_cooldown
			years = 1
		}

	}

	option = {
		name = gate_mana_saturation.2.b
		default_option = yes
	}
}

gate_mana_saturation.3 = {
	type = country_event
	dlc = dlc_magic_gate
	placement = scope:magic_growing_state
	category = gate_gate_mana_saturation

	title = gate_mana_saturation.3.t
	desc = gate_mana_saturation.3.d
	flavor = gate_mana_saturation.3.f

	event_image = {
		video = "africa_prosperous_farm"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = gfx/interface/icons/event_icons/event_gate_growing.dds

	duration = 3

	trigger = {
		exists = scope:magic_growing_state
	}

	immediate = {
		# Global cooldown between events
		set_global_variable = {
			name = gate_magic_soil_cooldown
			years = { 1 4 }
		}

		if = {
			limit = {
				has_variable = magic_growing_var
			}
			change_global_variable = {
				name = magic_growing_var
				add = 1
			}
		}
		else = {
			set_global_variable = {
				name = magic_growing_var
				value = 1
			}
		}

		# For debugging
		if = {
			limit = {
				NOT = {
					exists = scope:magic_growing_state
				}
			}
			random_scope_state = {
				save_scope_as = magic_growing_state
			}
		}
	}

	option = {
		name = gate_mana_saturation.3.a
		default_option = yes
		scope:magic_growing_state.state_region = {
			add_state_trait = state_trait_gate_magic_soil
			add_arable_land = 5
		}
		ai_chance = {
			base = 50
		}
	}
	option = {
		name = gate_mana_saturation.3.b
		scope:magic_growing_state.state_region = {
			add_state_trait = state_trait_gate_magic_soil
			add_arable_land = 5
		}
		hidden_effect = {
			trigger_event = {
				id = researching_magic.2
				months = 3
			}
		}
		gate_make_magic_researcher_busy = yes
		trigger = {
			has_gate_magic_researcher = yes
		}
		ai_chance = {
			base = 100
		}
	}
}
