namespace = gate_ley_line

gate_ley_line.1 = {
	type = country_event
	dlc = dlc_magic_gate
	placement = scope:gate_stonehenge_state

	title = gate_ley_line.1.t
	desc = gate_ley_line.1.d
	flavor = gate_ley_line.1.f

	event_image = {
		video = "gate_unspecified_obelisk"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_gate_nexus.dds"

	duration = 3

	trigger = {
		is_gate_magic_country = yes
		owns_entire_state_region = STATE_WEST_COUNTRY
		any_scope_state = {
			state_region ?= s:STATE_WEST_COUNTRY
			NOT = {
				has_state_trait = state_trait_gate_stonehenge
			}
		}
		s:STATE_WEST_COUNTRY ?= {
			var:mana_density >= gate_high_mana_density_threshhold
		}
	}

	immediate = {
		every_state = {
			limit = {
				state_region ?= s:STATE_WEST_COUNTRY
			}
			save_scope_as = gate_stonehenge_state
		}
	}

	option = {
		name = gate_ley_line.1.a
		s:STATE_WEST_COUNTRY ?= {
			add_state_trait = state_trait_gate_stonehenge
			hidden_effect = {
				add_arable_land = 10
			}
			gate_add_mana_density = {
				amount = 50
			}
		}
		trigger_event = {
			id = researching_magic.5
			months = 2
		}
	}
}
