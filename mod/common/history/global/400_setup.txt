# This scripts are run after all other history scripts were run
GLOBAL = {

	# Fixes colonization issues
	c:GATE_NS = {
		activate_law = law_type:law_gnomish_integration
	}
	c:GATE_AD = {
		activate_law = law_type:law_frontier_colonization
	}

	set_global_variable = {
		name = magic_growing_var
		value = 0
	}

	# Every country has a journal entry for mana saturation
	# since some countries can interact with the journal entry
	every_country = {
		limit = {
			NOT = {
				is_country_type = decentralized
			}
		}
		add_journal_entry = {
			type = je_mana_saturation
		}
	}

	# Make Mage Tower an owner building
	add_to_global_variable_list = {
		name = com_custom_owner_buildings
		target = bt:building_mage_tower
	}

	# Add custom buttons
	add_gate_magic_main_button = yes
	add_gate_magic_technology_button = yes

	set_global_variable = {
		name = magic_innovation_bank_max
		value = 30000
	}

	# Calculate magic innovation
	every_country = {
		set_variable = {
			name = magic_ethics
			value = 0
		}
		set_variable = {
			name = magic_innovation
			value = 0
		}
		calculate_magic_research = yes
	}

	# Statistics GUI
	gate_collect_magic_countries = yes

	every_in_global_list = {
		variable = magic_countries

		# Add magic researcher to all magic
		# countries
		create_character = {
			template = gate_magic_researcher_template
			interest_group = ig_gate_magic_users
		}
	}

	every_character = {
		limit = {
			OR = {
				culture ?= cu:dwarven
				culture ?= cu:elven
				culture ?= cu:vampire
			}
		}
		if = {
			limit = {
				culture ?= cu:dwarven
				NOT = {
					has_trait = gate_race_trait_dwarven
				}
			}
			add_trait = gate_race_trait_dwarven
		}
		if = {
			limit = {
				culture ?= cu:elven
				NOT = {
					has_trait = gate_race_trait_elven
				}
			}
			add_trait = gate_race_trait_elven
		}
		if = {
			limit = {
				culture ?= cu:vampire
				NOT = {
					has_trait = gate_race_trait_vampire
				}
			}
			add_trait = gate_race_trait_vampire
		}
	}

	every_state = {
		limit = { state_region = s:STATE_GABON }
		state_region = {
			add_arable_land = 25
		}
	}
	every_state = {
		limit = { state_region = s:STATE_CONGO_ORIENTALE }
		state_region = {
			add_arable_land = 30
		}
	}
	every_state = {
		limit = { state_region = s:STATE_KASAI }
		state_region = {
			add_arable_land = 30
		}
	}
	every_state = {
		limit = { state_region = s:STATE_CONGO }
		state_region = {
			add_arable_land = 35
		}
	}
	every_state = {
		limit = { state_region = s:STATE_EQUATEUR }
		state_region = {
			add_arable_land = 50
		}
	}
	every_state = {
		limit = { state_region = s:STATE_BAS_CONGO }
		state_region = {
			add_arable_land = 30
		}
	}
	every_state = {
		limit = { state_region = s:STATE_ZAMBIA }
		state_region = {
			add_arable_land = 40
		}
	}
	every_state = {
		limit = { state_region = s:STATE_KATANGA }
		state_region = {
			add_arable_land = 30
		}
	}
}
