# This scripts are run after all other history scripts were run
GLOBAL = {
	set_global_variable = {
		name = magic_growing_var
		value = 0
	}
	# Mana Saturation through gates.
	# Without further input will lead to full saturation in 1890s
	set_global_variable = {
		name = gate_mana_saturation_progress_var
		value = 1.5
	}
	calculate_monthly_mana_saturation = yes

	# Every country has a journal entry for mana saturation
	# since some countries can interact with the journal entry
	every_country = {
		limit = {
			NOT = {
				is_country_type = decentralized
			}
		}
		add_journal_entry = {
			type = je_mana_saturation
		}
	}

	add_mage_towers = yes

	# Set correct gate immigration traits
	set_immigration_traits = yes

	# Add Gate Button
	add_to_global_variable_list = {
		name = custom_button_list
		target = ideology:gate_button
	}

	# Add Gate Technology Button
	add_to_global_variable_list = {
		name = custom_button_list
		target = ideology:gate_button_technology
	}

	set_global_variable = {
		name = weekly_magic_innovation_base
		value = 25
	}
	every_country = {
		calculate_magic_research = yes
	}

	collect_magic_countries = yes
}
