gg_mr_has_magic_researcher_custom_loc = {
	type = country

	# From top to bottom decides which name list is appropriate.
	random_valid = no
	
	text = {
		trigger = {
			AND = {
				has_variable = gate_magic_researcher
				exists = var:gate_magic_researcher
				var:gate_magic_researcher = {
					is_character_alive = yes
				}
			}
		}
		localization_key = gg_mr_has_magic_researcher_gui_loc
	}

	text = {
		localization_key = gg_mr_has_no_magic_researcher_gui_loc
	}
}

gg_mr_magic_researcher_experience_custom_loc = {
	type = country

	# From top to bottom decides which name list is appropriate.
	random_valid = no

	text = {
		trigger = {
			NOT = {
				has_variable = gate_magic_researcher
			}
		}
		localization_key = gg_mr_no_magic_researcher_experience
	}

	text = {
		trigger = {
			var:gate_magic_researcher ?= {
				var:magic_experience >= gate_character_master_threshhold
			}
		}
		localization_key = gg_character_experience_expert
	}

	text = {
		trigger = {
			var:gate_magic_researcher ?= {
				var:magic_experience >= gate_character_experienced_threshhold
			}
		}
		localization_key = gg_character_experience_experienced
	}

	text = {
		localization_key = gg_character_experience_basic
	}
}
