mobilization_option_ritual_magic = {
	texture = "gfx/interface/icons/mobilization_options/gate_ritual_magic.dds"

	upkeep_modifier = {
		goods_input_mana_add = 3
	}

	unlocking_technologies = {
		ritual_magic
	}

	on_deactivate_while_mobilized = {
		custom_tooltip = {
			text = mobilization_option_it_hurts_morale_when_you_remove_supplies_while_in_combat_tt
			every_combat_unit = {
				add_morale = {
					value = morale
					multiply = -0.5
				}
			}
		}
	}
	
	unit_modifier = {
		unit_morale_loss_mult = -0.05
		unit_offense_mult = 0.1
		unit_defense_mult = 0.1
	}

	ai_weight = {
		value = 1
	}

	group = magical_support
}
