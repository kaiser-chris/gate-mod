gate_mana_saturation_progress_bar = {
 	name = "gate_mana_saturation_progress_bar_name"
 	desc = "gate_mana_saturation_progress_bar_desc"

	default = yes

 	yearly_progress = {
		if = {
			limit = {
				exists = global_var:gate_mana_saturation_progress_var
				exists = global_var:gate_mana_saturation_progress_mages_var
				exists = global_var:gate_mana_saturation_progress_buildings_var
			}
			add = {
			   desc = "gate_mana_saturation_progress_bar_balance"
			   value = global_var:gate_mana_saturation_progress_var
		    }
			add = {
				desc = "gate_mana_saturation_mages_progress_bar_balance"
				value = global_var:gate_mana_saturation_progress_mages_var
			}
			add = {
				desc = "gate_mana_saturation_buildings_progress_bar_balance"
				value = global_var:gate_mana_saturation_progress_buildings_var
			}
		}
 	}

	# We start at 7% because 1814 gates opened
	# and game starts in 1836
 	start_value = 7
 	min_value = 0
 	max_value = 100
}
