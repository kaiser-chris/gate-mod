	monthly_progress = {

		add = {
			desc = pb_gate_magic_research_base_value
			value = 1

		if = {
			limit = {
				has_modifier = focus_magic_research_cost_modifier

				desc = pb_gate_magic_research_focused_efforts
				value = 2
			}
		}
		# Magic Gate in a controlled State
		if = {
			limit = {
				has_building = building_gate
			}
			add = {
				desc = pb_gate_magic_research_magic_gate
				value = 2
			}
		}
	}

 	start_value = 0
 	min_value = 0
 	max_value = 100
