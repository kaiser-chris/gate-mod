pb_gate_german_integration_support = {
	name = "pb_gate_german_integration_support_name"
	desc = "pb_gate_german_integration_support_desc_pro"
	second_desc = "pb_gate_german_integration_support_desc_anti"

	# Every month check whether there are radicals or loyalists
	# Worst case = -3
	# Best case = +3
	monthly_progress = {
		every_scope_pop = {
			limit = {
				OR = {
					culture = cu:south_german
					culture = cu:north_german
					culture = cu:alemannic
				}
				pop_radical_fraction >= large_radicals
			}
			subtract = 1
		}
		every_scope_pop = {
			limit = {
				OR = {
					culture = cu:south_german
					culture = cu:north_german
					culture = cu:alemannic
				}
				pop_loyalist_fraction >= large_radicals
			}
			add = 1
		}
	}

	double_sided_bad = yes

	start_value = 60
	min_value = 0
	max_value = 100
}
