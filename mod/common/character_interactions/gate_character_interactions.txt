gate_promote_arcanist = {
    effect = {
        gate_activate_magic_researcher = yes
    }

    potential = {
        has_role = gate_arcanist
        home_country = {
            OR = {
                is_magic_country = yes
                has_journal_entry = je_gate_researching_magic
            }
        }
    }

    show_confirmation_box = yes

    cooldown = {
        years = 1
    }

    ai_will_do = {
        NOR = {
            has_variable = magic_researcher
            exists = var:magic_researcher
            var:magic_researcher ?= {
                is_character_alive = yes
            }
        }
    }

    ai_chance = {
        value = 100
    }

    should_ai_evaluate = {
        NOR = {
            has_variable = magic_researcher
            exists = var:magic_researcher
            var:magic_researcher ?= {
                is_character_alive = yes
            }
        }
    }

    possible = {
        has_role = gate_arcanist
        home_country = {
            custom_tooltip = {
                text = trigger_gate_is_not_magic_researcher
                NOT = {
                    var:magic_researcher ?= prev
                }
            }
            custom_tooltip = {
                text = magic_researcher_decision_actively_searching_tooltip
                NOT = {
                    has_variable = actively_hiring_magic_researcher
                }
            }
            has_active_magic_project = no
        }
    }
}
