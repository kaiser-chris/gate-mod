has_no_magic_research_alert = {
	script_context = player_country
	texture = gfx/interface/icons/alert_icons/has_no_magic_research.dds

	valid = {
		AND = {
			is_magic_country = yes
			NOT = {
				has_variable = magic_research
			}
		}
	}

	type = alert

	alert_group = gate_alerts

	color = green
}

has_not_started_research_of_magic_alert = {
	script_context = player_country
	texture = gfx/interface/icons/alert_icons/has_no_magic_research.dds

	valid = {
		AND = {
			is_magic_country = no
			NOT = {
				has_journal_entry = je_researching_magic
			}
		}
	}

	open_panel = journal

	type = alert

	alert_group = gate_alerts

	color = red
}

has_no_magic_researcher_alert = {
	script_context = player_country
	texture = gfx/interface/icons/alert_icons/has_no_magic_research.dds

	valid = {
		OR = {
			is_magic_country = yes
			has_journal_entry = je_researching_magic
		}
		NOT = {
			has_variable = magic_researcher
		}
	}

	open_panel = journal

	type = important_action

	alert_group = gate_alerts
}
