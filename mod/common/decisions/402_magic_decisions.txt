investigate_gate_decision = {
	is_shown = {
		NOT = {
			is_magic_country = yes
		}
		NOT = {
			is_country_type = decentralized
		}
	}

	possible = {
		is_diplomatic_play_committed_participant = no
		is_at_war = no
		is_magic_country = no
		custom_tooltip = {
			text = is_gate_planet_country_tt
			is_gate_planet_country = no
		}
		has_building = building_gate
		hidden_trigger = {
			NOT = { has_variable = je_investigate_gate_progress }
		}
	}

	when_taken = {
		set_variable = {
			name = je_investigate_gate_progress
			value = 0
		}
		add_journal_entry = {
			type = je_investigate_gate
		}
	}

	ai_chance = {
		base = 100
	}
}

research_magic_decision = {
	is_shown = {
		is_magic_country = no
		NOT = {
			has_journal_entry = je_researching_magic
		}
		NOT = {
			is_country_type = decentralized
		}
	}

	possible = {
		is_magic_country = no
		hidden_trigger = {
			NOT = { has_journal_entry = je_researching_magic }
		}
	}

	when_taken = {
		add_journal_entry = {
			type = je_researching_magic
		}
	}

	ai_chance = {
		base = 100
	}
}

magic_researcher_decision = {
	is_shown = {
		OR = {
			is_magic_country = yes
			has_journal_entry = je_researching_magic
		}
		NOT = {
			is_country_type = decentralized
		}
	}

	possible = {
		AND = {
			hidden_trigger = {
				OR = {
					is_magic_country = yes
					has_journal_entry = je_researching_magic
				}
			}
			custom_tooltip = {
				text = magic_researcher_decision_no_researcher_tooltip
				NOT = {
					has_variable = magic_researcher
				}
			}
		}
	}

	when_taken = {
		if = {
			limit = {
				is_magic_country = yes
			}
			create_character = {
				template = magic_researcher_template
				interest_group = ig_magic_users
				save_scope_as = magic_researcher
			}
		}
		else = {
			create_character = {
				template = magic_researcher_template
				interest_group = ig_intelligentsia
				save_scope_as = magic_researcher
			}
		}
		hidden_effect = {
			scope:magic_researcher = {
				place_character_in_void = 10000
			}
			set_variable = {
				name = magic_researcher
				value = scope:magic_researcher
			}
			if = {
				limit = {
					has_journal_entry = je_researching_magic
				}
				je:je_researching_magic = {
					clear_variable_list = com_journal_characters
					add_to_variable_list = {
						name = com_journal_characters
						target = scope:magic_researcher
					}
				}
			}
		}
	}

	ai_chance = {
		base = 100
	}
}
