debug_mana_saturation_decision = {
	is_shown = {
		has_global_variable = gate_debug
		is_player = yes
		NOT = { has_global_variable = gate_mana_saturation_full }
	}

	when_taken = {
		set_global_variable = {
			name = gate_mana_saturation_full
		}
		set_global_variable = {
			name = global_mana_saturation
			value = 100
		}
	}

	ai_chance = {
		base = 0
	}
}
