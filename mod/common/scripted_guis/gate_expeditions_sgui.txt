sgui_gate_map_choose_location = {
    scope = character

    effect = {
        if = {
            limit = {
                global_var:gate_map_sidebar ?= root
                has_global_variable = gate_map_sidebar
            }
            remove_global_variable = gate_map_sidebar
        }
        else = {
            set_global_variable = {
                name = gate_map_sidebar
                value = root
            }
        }
    }
}

sgui_gate_map_reset_location = {
    scope = none

    effect = {
        remove_global_variable = gate_map_sidebar
    }
}
