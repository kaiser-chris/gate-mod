sgui_gate_close_window = {
	scope = country

	effect = {
		remove_variable = gate_main_window_is_active
		remove_variable = gate_main_window_active_tab
		remove_variable = gate_main_technology_is_active
		remove_variable = gate_main_technology_active_tab
	}
}

sgui_gate_window_visisblity = {
	scope = country

	is_shown = {
		AND = {
			has_variable = gate_main_window_is_active
			has_variable = gate_main_window_active_tab
		}
	}
}

sgui_gate_window_tab_magic = {
	scope = country

	is_shown = {
		AND = {
			has_variable = gate_main_window_is_active
			has_variable = gate_main_window_active_tab
			var:gate_main_window_active_tab = 1
		}
	}

	effect = {
		set_variable = {
			name = gate_main_window_active_tab
			value = 1
		}
	}
}

sgui_gate_window_tab_research = {
	scope = country

	is_shown = {
		AND = {
			has_variable = gate_main_window_is_active
			has_variable = gate_main_window_active_tab
			var:gate_main_window_active_tab = 2
		}
	}

	effect = {
		set_variable = {
			name = gate_main_window_active_tab
			value = 2
		}
	}
}

sgui_gate_enable_debug = {
	scope = country

	effect = {
		change_infamy = 10
		set_global_variable = {
			name = gate_debug
			value = 1
		}
	}
}
