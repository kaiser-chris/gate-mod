    scope = country

    is_valid = {
        gate_project_can_start_improve_mana_extraction = yes
        gate_project_has_any_selector_open = no
    }

    effect = {
        every_scope_building = {
            limit = {
                is_building_group = bg_gate_mana_extraction
                NOT = {
                    has_modifier = modifier_gate_project_improved_mana_extraction
                }
            }
            root = {
                add_to_variable_list = {
                    name = gate_project_improve_mana_extraction_selector
                    target = prev
                }
            }
        }
    }
}
