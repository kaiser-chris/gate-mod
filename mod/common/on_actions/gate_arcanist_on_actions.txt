# Root = Character
on_yearly_pulse_character = {
    on_actions = {
        on_yearly_gate_magic_researcher_experience
    }
}

on_yearly_gate_magic_researcher_experience = {
    trigger = {
        OR = {
            has_role = gate_arcanist
            has_role = gate_active_arcanist
        }
    }
    effect = {
        gate_calculate_researcher_experience = yes
    }
}

# Root = Character
on_character_death = {
    on_actions = {
        on_gate_magic_researcher_death
    }
}

on_gate_magic_researcher_death = {
    trigger = {
        has_role = gate_active_arcanist
    }
    effect = {
        save_scope_as = magic_researcher
        owner = {
            post_notification = message_gate_arcanist_died
        }
    }
}

on_game_started_after_lobby = {
    on_actions = {
        on_game_started_randomize_hiring_dates
    }
}

# Randomize arcanist hiring date
# so they are not 0 at game start
on_game_started_randomize_hiring_dates = {
    effect = {
        every_country = {
            var:gate_magic_researcher ?= {
                set_variable = {
                    name = gate_hired
                    value = {
                        value = game_date
                        subtract = { 60 90 }
                    }
                }
            }
        }
    }
}
