on_game_started = {
    on_actions = {
        on_gate_game_setup
    }
}

on_gate_game_setup = {
    effect = {
        # Statistics GUI
        gate_collect_magic_countries = yes

        every_in_global_list = {
            variable = magic_countries
            limit = {
                NOT = {
                    # Exclude Texas since it will lead
                    # to issues. This is handled in:
                    # 350_base_game_extension.txt
                    c:TEX ?= this
                }
            }

            # Add magic researcher to all magic
            # countries
            gate_create_and_activate_magic_researcher = yes
        }

        # Add race traits to all
        # automatically generated characters
        gate_add_race_traits = yes

        # Add arable land for balancing
        gate_setup_arable_land = yes

        c:CHI = {
            # Great Qing
            # Add Magic Researcher
            gate_create_and_activate_magic_researcher = yes
        }
        c:KOR = {
            # Korea
            # Add Magic Researcher
            gate_create_and_activate_magic_researcher = yes
        }
        c:GBR = {
            # Great Britain
            # Add Magic Researcher
            gate_create_and_activate_magic_researcher = yes
        }
        c:POR = {
            # Portugal
            # Add Magic Researcher
            gate_create_and_activate_magic_researcher = yes
        }
        c:MEX = {
            # Mexico
            # Add Magic Researcher
            gate_create_and_activate_magic_researcher = yes
        }
        c:USA = {
            # United States
            # Add Magic Researcher
            gate_create_and_activate_magic_researcher = yes
        }
        c:TEX = {
            # Texas
            # Add Magic Researcher
            create_character = {
                template = gate_magic_researcher_template
                interest_group = ig_intelligentsia
                culture = gate_south_ogre
                save_scope_as = magic_researcher_south_ogre
            }
            scope:magic_researcher_south_ogre = {
                gate_activate_magic_researcher = yes
            }
        }

        # Starting literacy for gate countries and races
        c:GATE_AD = {
            effect_starting_pop_literacy_low = yes
            effect_starting_pop_literacy_gate = yes
        }
        c:GATE_MK = {
            effect_starting_pop_literacy_high = yes
            effect_starting_pop_literacy_gate = yes
        }
        c:GATE_OK = {
            effect_starting_pop_literacy_low = yes
            effect_starting_pop_literacy_gate = yes
        }
        c:GATE_NS = {
            effect_starting_pop_literacy_middling = yes
            effect_starting_pop_literacy_gate = yes
        }
        c:GATE_BN = {
            effect_starting_pop_literacy_low = yes
            effect_starting_pop_literacy_gate = yes
        }

        # Making 100% sure self requiring PM is set
        every_country = {
            limit = {
                capital = {
                    has_building = building_gate_magic_research_institute
                }
            }
            activate_production_method = {
                building_type = building_gate_magic_research_institute
                production_method = pm_gate_no_researcher
            }
            gate_reset_research_institute_pm = yes
        }
    }
}
