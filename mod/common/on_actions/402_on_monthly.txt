# Root = any
on_monthly_pulse = {
    on_actions = {
        on_monthly_gate_calculate_mana_saturation
        on_monthly_gate_statistics
        on_monthly_gate_mana_density
        on_monthly_ui_check
    }
}

on_monthly_ui_check = {
    effect = {
        add_gate_magic_main_button = yes
        if = {
            limit = {
                NOT = {
                    any_in_global_list = {
                        variable = com_topbar_items
                        this = flag:gg_topbar_mana_saturation
                    }
                }
            }
            add_com_topbar_element = {
                element_name = gg_topbar_mana_saturation
            }
            every_country = {
                add_to_variable_list = {
                    name = com_topbar_second_line
                    target = scope:gg_topbar_mana_saturation
                }
            }
        }
        if = {
            limit = {
                NOT = {
                    any_in_global_list = {
                        variable = com_topbar_items
                        this = flag:gg_topbar_magic_innovation
                    }
                }
            }
            add_com_topbar_element = {
                element_name = gg_topbar_magic_innovation
            }
            every_country = {
                add_to_variable_list = {
                    name = com_topbar_second_line
                    target = scope:gg_topbar_magic_innovation
                }
            }
        }
    }
}

on_monthly_gate_calculate_mana_saturation = {
    trigger = {
        global_var:global_mana_saturation <= 100
        OR = {
            # Every four months
            month = 0
            month = 4
            month = 8
        }
    }
    effect = {
        gate_calculate_mana_saturation = yes
    }
}

on_monthly_gate_statistics = {
    effect = {
        gate_collect_magic_countries = yes
    }
}

on_monthly_gate_mana_density = {
    trigger = {
        global_var:global_mana_saturation <= 100
        OR = {
            # Every three months
            month = 0
            month = 3
            month = 6
            month = 9
        }
    }
    effect = {
        set_local_variable = {
            name = gate_mana_propagation_amount
            value = { 0 2 }
        }
        gate_propogate_mana_density = {
            amount = local_var:gate_mana_propagation_amount
        }
    }
}

on_monthly_pulse_state = {
    on_actions = {
        on_monthly_gate_mana_density_modifier
    }
}

on_monthly_gate_mana_density_modifier = {
    trigger = {
        modifier:state_gate_mana_density_increase_add != 0
        OR = {
            # Every three months
            month = 0
            month = 3
            month = 6
            month = 9
        }
    }
    effect = {
        state_region = {
            gate_change_mana_density = {
                amount = root.modifier:state_gate_mana_density_increase_add
            }
        }
    }
}

# Root = country
on_monthly_pulse_country = {
    on_actions = {
        on_monthly_gate_add_immigration_dwarven
        on_monthly_gate_add_immigration_orcish
        on_monthly_gate_add_immigration_elven
        on_monthly_gate_add_immigration_gnomish
        on_monthly_gate_add_immigration_bear
    }
    random_events = {
        chance_of_no_event = 80
        1 = gate_arcane_council.1
        1 = gate_arcane_council.2
    }
    events = {
        gate_pelly.1
    }
}
