		is_diplomatic_play_committed_participant = no
		is_at_war = no
		custom_tooltip = {
			text = je_dwarven_alps_button_switzerland_exists
			exists = c:SWI
		}
		c:SWI ?= {
			custom_tooltip = {
				text = je_dwarven_alps_button_switzerland_magic_already
				NOT = {
					has_variable = swiss_magic_var

				OR = {

						}
					}
				}
			}
		}
	}

	ai_chance = {
		value = 100
	}

	effect = {
		trigger_event = {
			id = dwarven_alps.1
			popup = yes
		}
