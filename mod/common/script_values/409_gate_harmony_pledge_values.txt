gate_hp_society_dist_of_power = {
	if = {
		limit = { # Absolute Power / No Power
			OR = { 
				has_law_or_variant = law_type:law_autocracy
				has_law_or_variant = law_type:law_single_party_state
			}
		}
		add = 50
	}
	else_if = {
		limit = { # Elite Captured Power
			OR = { 
				has_law_or_variant = law_type:law_oligarchy
				has_law_or_variant = law_type:law_elder_council
			}
		}
		add = 40
	}
	else_if = {
		limit = { has_law_or_variant = law_type:law_technocracy  }
		add = 30
	}
	else_if = {
		limit = { has_law_or_variant = law_type:law_landed_voting }
		add = 20
	}
	else_if = {
		limit = { has_law_or_variant = law_type:law_census_voting }
		add = 0
	}
	else_if = {
		limit = { has_law_or_variant = law_type:law_universal_suffrage }
		add = -20
	}
	else_if = {
		limit = { has_law_or_variant = law_type:law_anarchy }
		add = -40
	}
	else = {		# For random modded laws
		add = gate_hp_society_dist_of_power_modded
	}
	divide = -100
}
gate_hp_society_dist_of_power_modded = 0	# Replace this value for modded laws compatibility

gate_hp_society_internal_security = {
	if = {
		limit = { has_law_or_variant = law_type:law_secret_police }
		add = {
			value = 0.15		# Secret police that aren't strongly *felt* provide many opportunities to the inventive..
			desc = GATE_HP_BASE_LAW
			format = GATE_HP_SCRIPT_VALUE_FORMAT
		}
		add = {
			add = -0.10	# But, the stronger this institution is, the worse stratification gets...
			multiply = {
				value = institution:institution_home_affairs.investment
			}
			desc = GATE_HP_INSTITUTION_LEVEL
			format = GATE_HP_SCRIPT_VALUE_FORMAT
		}
	}
	else_if = {
		limit = { has_law_or_variant = law_type:law_national_guard }
		add = {
			value = -0.20
			desc = GATE_HP_BASE_LAW
			format = GATE_HP_SCRIPT_VALUE_FORMAT
		}
		add = {			# Using the investment level as a proxy for level of societal involvement:
			add = 0.05	# NG existing adds stratification, higher investment breeds civic engagement to counter
			multiply = {
				value = institution:institution_home_affairs.investment
			}
			desc = GATE_HP_INSTITUTION_LEVEL
			format = GATE_HP_SCRIPT_VALUE_FORMAT
		}
	}
	else_if = {
		limit = { has_law_or_variant = law_type:law_guaranteed_liberties }
		add = {
			value = 0.05
			desc = GATE_HP_BASE_LAW
			format = GATE_HP_SCRIPT_VALUE_FORMAT
		}
		multiply = {
			value = institution:institution_home_affairs.investment
			desc = GATE_HP_INSTITUTION_LEVEL
			format = GATE_HP_SCRIPT_VALUE_FORMAT
		}
	}
	else_if = {
		limit = { has_law_or_variant = law_type:law_no_home_affairs }
		add = 0.30
	}
	else = {	# For Random Modded laws
		add = {
			value = gate_hp_society_internal_security_modded
			owner.institution:institution_home_affairs ?= {	# for modded variants with an institution
				desc = GATE_HP_BASE_LAW
				format = GATE_HP_SCRIPT_VALUE_FORMAT
			}
		}
		owner.institution:institution_home_affairs ?= {			
			add = {
				value = investment
				multiply = gate_hp_society_internal_security_investment_modded
				desc = GATE_HP_INSTITUTION_LEVEL
				format = GATE_HP_SCRIPT_VALUE_FORMAT
			}
		}
	}
}

gate_harmony_pledge_society_trend_from_demographics = {	# +ve = antinomian (chaos), -ve = stratified (order)
	value = 0
	if = {
		limit = { has_law = law_type:law_gate_gnomish_harmony }
		add = {
			add = {
				value = 0.5
				multiply = var:gate_hp_gnomish_percent		# gnomes/total				# More gnomes = more chaos
				desc = GATE_HP_GNOMISH_POP_PERCENT
				format = GATE_HP_SCRIPT_VALUE_FORMAT	# (optional)
			}
			add = {
				value = -1.0
				multiply = var:gate_hp_non_gnomish_percent
				desc = GATE_HP_NON_GNOME_POP_PERCENT
				format = GATE_HP_SCRIPT_VALUE_FORMAT	# (optional)
			}
			multiply = {		# More pledge adherents = more regression towards harmony
				value = 1		# Above value is *reduced* proportionally i.e. 100% pledge adoption = multiply by 0
				subtract = var:gate_hp_pledge_percent
				desc = GATE_HP_PLEDGE_PERCENT
				format = GATE_HP_SCRIPT_VALUE_FORMAT	# (optional)
			}
		}				# Net effect - large % of non-gnomes who are also non-pledgers will destabilise.
	}
}

	# Center drift modifiers
	if = {
		limit = {
			scope:journal_entry = {
				scripted_bar_progress = {
					name = pb_gate_harmony_pledge_society
					value >= 50
				}
			}
			modifier:country_gate_harmony_pledge_society_drift_center != 0

gate_harmony_pledge_economy_trend_scaling = {	# works against movement towards disharmony
	if = {
		limit = {
			gate_harmony_pledge_economy_trend_base < 0
			"je:je_gate_harmony_pledge.scripted_bar_progress(pb_gate_harmony_pledge_economy)" < 50
		}
		add = {
			add = gate_harmony_pledge_economy_trend_base
			multiply = {
				add = "je:je_gate_harmony_pledge.scripted_bar_progress(pb_gate_harmony_pledge_economy)"
				divide = 50
			}
			subtract = gate_harmony_pledge_economy_trend_base
			# Base SQ resistance unrelated to speed of change.
			if = {
				limit = { "je:je_gate_harmony_pledge.scripted_bar_progress(pb_gate_harmony_pledge_economy)" < 40 }
				add = {
					add = 40
					subtract = "je:je_gate_harmony_pledge.scripted_bar_progress(pb_gate_harmony_pledge_economy)"
					divide = 100
				}
			}
			multiply = var:gate_hp_non_gnomish_percent
			desc = GATE_HP_NATURAL_CHANGE_RESISTANCE
			format = GATE_HP_SCRIPT_VALUE_FORMAT	# (optional)
		}
	}
	else_if = {
		limit = {
			gate_harmony_pledge_economy_trend_base > { value = 0 }
			"je:je_gate_harmony_pledge.scripted_bar_progress(pb_gate_harmony_pledge_economy)" > 50
		}
		add = {
			add = gate_harmony_pledge_economy_trend_base
			multiply = {
				add = 100
				subtract = "je:je_gate_harmony_pledge.scripted_bar_progress(pb_gate_harmony_pledge_economy)"
				divide = 50
			}
			subtract = gate_harmony_pledge_economy_trend_base
			# Base SQ resistance unrelated to speed of change.
			if = {
				limit = { "je:je_gate_harmony_pledge.scripted_bar_progress(pb_gate_harmony_pledge_economy)" > 50 }
				add = {
					add = 60
					subtract = "je:je_gate_harmony_pledge.scripted_bar_progress(pb_gate_harmony_pledge_economy)"
					divide = 200
				}
			}
			multiply = var:gate_hp_non_gnomish_percent
			desc = GATE_HP_NATURAL_CHANGE_RESISTANCE
			format = GATE_HP_SCRIPT_VALUE_FORMAT	# (optional)
		}
	}
	multiply = {
		value = 1
		add = var:gate_hp_pledge_percent
		desc = GATE_HP_PLEDGE_PERCENT_SCALING
		format = GATE_HP_SCRIPT_VALUE_FORMAT	# (optional)
	}
}
