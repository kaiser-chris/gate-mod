gate_hp_society_dist_of_power = {
	if = {
		limit = { # Absolute Power / No Power
			OR = { 
				has_law_or_variant = law_type:law_autocracy
				has_law_or_variant = law_type:law_single_party_state
			}
		}
		add = 50
	}
	else_if = {
		limit = { # Elite Captured Power
			OR = { 
				has_law_or_variant = law_type:law_oligarchy
				has_law_or_variant = law_type:law_elder_council
			}
		}
		add = 40
	}
	else_if = {
		limit = { has_law_or_variant = law_type:law_technocracy  }
		add = 30
	}
	else_if = {
		limit = { has_law_or_variant = law_type:law_landed_voting }
		add = 20
	}
	else_if = {
		limit = {
			has_law_or_variant = law_type:law_technocracy
		}
		add = 10
	}
	else_if = {
		limit = { has_law_or_variant = law_type:law_census_voting }
		add = 0
	}
	else_if = {
		limit = { has_law_or_variant = law_type:law_universal_suffrage }
		add = -20
	}
	else_if = {
		limit = { has_law_or_variant = law_type:law_anarchy }
		add = -40
	}
	else = {		# For random modded laws
		add = gate_hp_society_dist_of_power_modded
	}
}
gate_hp_society_dist_of_power_modded = 0	# Replace this value for modded laws compatibility

gate_hp_society_internal_security = {
	if = {
		limit = { has_law_or_variant = law_type:law_secret_police }
		add = -15		# Secret police that aren't strongly *felt* provide many opportunities to the inventive..
		add = {
			add = 10	# But, the stronger this institution is, the worse stratification gets...
			multiply = {
				value = institution:institution_home_affairs.investment
			}
		}
	}
	else_if = {
		limit = { has_law_or_variant = law_type:law_national_guard }
		add = 20
		add = {			# Using the investment level as a proxy for level of societal involvement:
			add = -5	# NG existing adds stratification, higher investment breeds civic engagement to counter
			multiply = {
				value = institution:institution_home_affairs.investment
			}
		}
	}
	else_if = {
		limit = { has_law_or_variant = law_type:law_guaranteed_liberties }
		add = -5
		multiply = {
			value = institution:institution_home_affairs.investment
		}
	}
	else_if = {
		limit = { has_law_or_variant = law_type:law_no_home_affairs }
		add = -30
	}
	else = {	# For Random Modded laws
		add = gate_hp_society_internal_security_modded
	}
}
gate_hp_society_internal_security_modded = 0	# Replace this value for modded laws compatibility

gate_hp_society_education = {
	if = {
		limit = {
			OR = { 
				has_law_or_variant = law_type:law_religious_schools
				has_law_or_variant = law_type:law_private_schools
			}
		}
		add = 5
		multiply = {
			value = institution:institution_schools.investment	# level 1 = 0
			add = 1
		}
	}
	else_if = {
		limit = { has_law_or_variant = law_type:law_no_schools }
		add = 15
	}
	else_if = {
		limit = { has_law_or_variant = law_type:law_public_schools }
		add = -2
		multiply = {
			value = institution:institution_schools.investment
			add = 1
		}
	}
	else = {	# For Random Modded laws
		add = gate_hp_society_education_modded
		owner.institution:institution_schools ?= {
			multiply = {
				value = investment	# institution:institution_schools.investment # already in institution scope
				add = 1				# level 1 = 0
			}
		}
	}
}
gate_hp_society_education_modded = 0	# Replace this value for modded laws compatibility

gate_hp_society_citizenship = {
	value = 0
	if = {
		limit = { has_law = law_type:law_gate_gnomish_harmony }
		add = {
			add = {
				value = -50
				multiply = {								# More gnomes = more chaos
					value = var:gate_hp_gnomish_percent
					divide = var:gate_hp_total_population
				}
			}
			add = {
				value = 100
				multiply = {								# More non-gnomes = more order
					value = var:gate_hp_citizenship_percent
					subtract = var:gate_hp_gnomish_percent
					divide = var:gate_hp_total_population
				}
			}
			add = -50			# Temporarily set "harmonious" = 0
			multiply = {		# More pledge adherents = more regression towards harmony
				value = 1
				subtract = {
					value = var:gate_hp_pledge_percent
					divide = var:gate_hp_total_population
				}
			}
			add = 50			# Restore "harmonious" = 50
		}				# Net effect - large % of non-gnomes who are also non-pledgers will destabilise.
	}
}
