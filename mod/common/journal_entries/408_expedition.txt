je_gate_expedition_reytu_preparation = {
	icon = "gfx/interface/icons/event_icons/event_map.dds"

	group = je_group_expeditions

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_gate_expedition_reytu_preparation_hiring_leader
				trigger = {
					has_variable = expedition_reytu_preparation_hiring_leader
					NOT = {
						var:expedition_reytu ?= {
							has_variable = leader
						}
					}
				}
			}
			triggered_desc = {
				desc = je_gate_expedition_reytu_preparation_in_progess
			}
		}
	}

	immediate = {
		set_variable = {
			name = gate_expedition_preparation_progress
			value = 0
		}
		set_variable = {
			name = gate_expedition_preparation_max
			value = var:expedition_reytu.var:expedition_type
		}
	}

	complete = {
		custom_tooltip = {
			text = je_gate_expedition_reytu_preparation_complete_tooltip
			var:gate_expedition_preparation_progress = var:gate_expedition_preparation_max
		}
	}

	fail = {
		custom_tooltip = {
			text = je_gate_expedition_reytu_preparation_fail_tooltip
			NOT = {
				has_building = building_gate_alxa
			}
			has_variable = gate_expedition_cancel
		}
	}

	on_complete = {
		hidden_effect = {
			var:expedition_reytu ?= {
				set_variable = expedition_started
			}
		}
	}

	on_fail = {
		hidden_effect = {
			remove_variable = gate_expedition_preparation_progress
			remove_variable = gate_expedition_preparation_max
			remove_variable = gate_expedition_cancel
		}
	}

	scripted_button = je_gate_expedition_reytu_preparation_button_hire_leader
	scripted_button = je_gate_expedition_reytu_preparation_button_start_preparations
	scripted_button = je_gate_expedition_reytu_preparation_button_stop_preparations
	scripted_button = je_gate_expedition_reytu_preparation_button_open_map
	scripted_button = je_gate_expedition_reytu_preparation_button_cancel

	progressbar = yes
	current_value = {
		value = root.var:gate_expedition_preparation_progress
	}
	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					has_variable = gate_expedition_preparation_active
				}
				change_variable = {
					name = gate_expedition_preparation_progress
					add = 1
					min = 0
					max = var:gate_expedition_preparation_max
				}
			}
		}
	}

	transferable = no
	can_revolution_inherit = yes

	weight = 200
	should_be_pinned_by_default = yes
}
