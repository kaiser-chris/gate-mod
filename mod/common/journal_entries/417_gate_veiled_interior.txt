je_gate_veiled_interior = {
    group = je_group_internal_affairs
    icon = "gfx/interface/icons/event_icons/event_map.dds"
    weight = 100

    widget = {
        gui = "gui/com_journal_injects/injects.gui"
        name = "com_journal_entry_enactment"
        container = "custom_widget_container_2"
    }

    widget = {
        gui = "gui/journal_injects/gate_veiled_interior_injects.gui"
        name = "gate_inject_veiled_interior_phase"
        container = "custom_widget_container_4"
    }

    widget = {
        gui = "gui/com_journal_injects/injects.gui"
        name = "com_journal_entry_dlc_icon"
        container = "custom_widget_container_7"
    }

    is_shown_in_lobby = {
        c:GATE_NS ?= this
    }

    scripted_button = sb_gate_veiled_interior_expedition
    scripted_button = sb_gate_veiled_interior_scout

    status_desc = {
        first_valid = {
            triggered_desc = {
                trigger = {
                    scope:journal_entry = {
                        com_is_stalled_enactment_journal = yes
                    }
                }
                desc = je_gate_veiled_interior_status_diplomacy_stall
            }
            triggered_desc = {
                trigger = {
                    scope:journal_entry = {
                        com_is_phase_enactment_journal = {
                            phase = je_gate_veiled_interior_diplomacy_discussions
                        }
                    }
                }
                desc = je_gate_veiled_interior_status_diplomacy_discussions
            }
            triggered_desc = {
                trigger = {
                    scope:journal_entry = {
                        com_is_phase_enactment_journal = {
                            phase = je_gate_veiled_interior_diplomacy_alignment
                        }
                    }
                }
                desc = je_gate_veiled_interior_status_diplomacy_alignment
            }
            triggered_desc = {
                trigger = {
                    scope:journal_entry = {
                        com_is_phase_enactment_journal = {
                            phase = je_gate_veiled_interior_diplomacy_implementation
                        }
                    }
                }
                desc = je_gate_veiled_interior_status_diplomacy_implementation
            }
            triggered_desc = {
                trigger = {
                    scope:journal_entry = {
                        has_variable = gate_veiled_interior_expedition
                    }
                }
                desc = je_gate_veiled_interior_status_expedition
            }
            triggered_desc = {
                trigger = {
                    scope:journal_entry = {
                        has_variable = gate_veiled_interior_scouting
                    }
                }
                desc = je_gate_veiled_interior_status_scouting
            }
            triggered_desc = {
                trigger = {
                    scope:journal_entry = {
                        var:gate_veiled_interior_phase ?= flag:je_gate_veiled_interior_phase_veiled
                    }
                }
                desc = je_gate_veiled_interior_status_veiled
            }
            triggered_desc = {
                trigger = {
                    scope:journal_entry = {
                        var:gate_veiled_interior_phase ?= flag:je_gate_veiled_interior_phase_early_contact
                    }
                }
                desc = je_gate_veiled_interior_status_early_contact
            }
        }
    }

    immediate = {
        add_gate_dlc_icon = yes
        scope:journal_entry = {
            set_variable = {
                name = gate_veiled_interior_phase
                value = flag:je_gate_veiled_interior_phase_veiled
            }
        }
    }

    on_weekly_pulse = {
        trigger = {
            scope:journal_entry = {
                var:gate_veiled_interior_phase ?= flag:je_gate_veiled_interior_phase_diplomacy
                com_is_stalled_enactment_journal = no
            }
            num_pending_events:gate_veiled_interior_enactment = 0
        }
        effect = {
            scope:journal_entry = {
                set_local_variable = {
                    name = new_step_progress
                    value = {
                        value = com_enactment_journal_step_progress
                        add = { 0.05 0.15 }
                    }
                }
                clamp_local_variable = {
                    name = new_step_progress
                    min = 0
                    max = 1
                }
                com_set_enactment_journal_step_progress = {
                    value = local_var:new_step_progress
                }

                if = {
                    limit = {
                        local_var:new_step_progress >= 1
                    }
                    set_local_variable = {
                        name = chance
                        value = { 0 1 }
                    }
                    if = {
                        # Stalled
                        limit = {
                            local_var:chance <= com_enactment_journal_stall_chance
                        }
                        root = {
                            trigger_event = {
                                on_action = on_gate_veiled_interior_diplomacy_stall
                            }
                        }
                    }
                    else_if = {
                        # Success
                        limit = {
                            local_var:chance >= {
                                value = 1
                                subtract = com_enactment_journal_success_chance
                            }
                        }
                        root = {
                            trigger_event = {
                                on_action = on_gate_veiled_interior_diplomacy_success
                            }
                        }
                    }
                    else = {
                        # Nothing
                        root = {
                            trigger_event = {
                                on_action = on_gate_veiled_interior_diplomacy_nothing
                            }
                        }
                    }
                    com_set_enactment_journal_step_progress = {
                        value = 0
                    }
                    gate_veiled_interior_enactment_calculate_chances = yes
                }
            }
        }
    }

    invalid = {
        NOT = {
            any_scope_state = {
                OR = {
                    state_region = s:STATE_EASTERN_NEW_GUINEA
                    state_region = s:STATE_WESTERN_NEW_GUINEA
                }
            }
        }
    }

    should_be_pinned_by_default_involved = yes
    should_be_pinned_by_default_uninvolved_or_context = yes
}
