je_gate_vampire_emergence = {
	icon = gfx/interface/icons/religion_icons/gate_opera_mask.dds

	group = je_group_internal_affairs

	scripted_button = je_gate_vampire_emergence_button_decapitate
	scripted_button = je_gate_vampire_emergence_button_purge

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_gate_vampire_emergence_decapitated
				trigger = {
					has_variable = gate_vampire_decapitated
				}
			}
			triggered_desc = {
				desc = je_gate_vampire_emergence_truth_known
				trigger = {
					has_variable = gate_vampire_truth
				}
			}
			triggered_desc = {
				desc = je_gate_vampire_emergence_allure
			}
		}
	}

	immediate = {
        scope:journal_entry = {
            add_com_dlc_icon = {
                dlc = dlc_magic_gate
            }
        }
		save_scope_as = gate_vampire_country
	}

	on_yearly_pulse = {
		# Convert 1-2% of pops to vampires every year
		effect = {
			if = {
				limit = {
					OR = {
						has_variable = gate_vampire_truth
						has_variable = gate_vampire_decapitated
					}
				}
				every_scope_pop = {
					limit = {
						NOR = {
							is_pop_type = clergymen
							is_pop_type = slaves
						}
					}
					change_pop_culture = {
						target = cu:vampire
						value = 0.01
					}
				}
			}
			else = {
				every_scope_pop = {
					limit = {
						NOR = {
							is_pop_type = clergymen
							is_pop_type = slaves
						}
					}
					change_pop_culture = {
						target = cu:vampire
						value = 0.02
					}
				}
			}
			every_scope_pop = {
				limit = {
					culture = cu:vampire
				}
				change_pop_religion = {
					target = rel:gate_lunar_masquerade
					value = 1
				}
			}
		}
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					NOT = {
						has_variable = gate_vampire_decapitated
					}
				}
				je:je_gate_vampire_emergence ?= {
					clear_variable_list = com_journal_characters
					add_to_variable_list = {
						name = com_journal_characters
						target = root.var:gate_character_vampire
					}
				}
			}
		}
		# Trigger vampire events
		random_events = {
			chance_of_no_event = 80
			1 = gate_vampire.5
			1 = gate_vampire.6
			1 = gate_vampire.7
			1 = gate_vampire.8
			1 = gate_vampire.11
			1 = gate_vampire.12
			1 = gate_vampire.13
			1 = gate_vampire.14
			1 = gate_vampire.15
			1 = gate_vampire.16
		}
	}

	possible = {
		custom_tooltip = {
			text = je_gate_vampire_emergence_activate
			has_variable = gate_character_vampire
		}
	}

	complete = {
		custom_tooltip = {
			text = je_gate_vampire_emergence_complete
			has_law = law_type:law_gate_vampire_monarchy
		}
	}

	fail = {
		custom_tooltip = {
			text = je_gate_vampire_emergence_fail
			has_variable = gate_rejected_vampirism
		}
	}

	invalid = {
		custom_tooltip = {
			text = je_gate_vampire_emergence_invalid
			has_variable = gate_vampire_decapitated
			NOT = {
				has_variable = gate_vampire_wait
			}
		}
	}

	on_complete = {
		hidden_effect = {
			trigger_event = {
				id = gate_vampire.1
				popup = yes
			}
		}
	}

	on_invalid = {
		hidden_effect = {
			trigger_event = {
				id = gate_vampire.10
				popup = yes
			}
			remove_variable = gate_vampire_decapitated
			remove_variable = gate_vampire_truth
		}
	}

	on_fail = {
		hidden_effect = {
			trigger_event = {
				id = gate_vampire.2
				popup = yes
			}
		}
	}

	progressbar = no

	transferable = no
	can_revolution_inherit = yes

	weight = 200
	should_be_pinned_by_default = yes
}
