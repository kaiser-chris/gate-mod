je_gate_vampire_emergence = {
	icon = gfx/interface/icons/religion_icons/gate_opera_mask.dds

	group = je_group_internal_affairs

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_gate_vampire_emergence_truth_known
				trigger = {
					has_variable = gate_vampire_truth
				}
			}
			triggered_desc = {
				desc = je_gate_vampire_emergence_allure
			}
		}
	}

	immediate = {
		save_scope_as = gate_vampire_country
	}

	on_yearly_pulse = {
		# Convert 1-2% of pops to vampires every year
		effect = {
			if = {
				limit = {
					has_variable = gate_vampire_truth
				}
				every_scope_pop = {
					limit = {
						NOR = {
							is_pop_type = clergymen
							is_pop_type = slaves
						}
					}
					change_pop_culture = {
						target = cu:vampire
						value = 0.01
					}
				}
			}
			else = {
				every_scope_pop = {
					limit = {
						NOR = {
							is_pop_type = clergymen
							is_pop_type = slaves
						}
					}
					change_pop_culture = {
						target = cu:vampire
						value = 0.02
					}
				}
			}
			every_scope_pop = {
				limit = {
					culture = cu:vampire
				}
				change_pop_religion = {
					target = rel:lunar_masquerade
					value = 1
				}
			}
		}
		# Trigger vampire events
		random_events = {
			chance_of_no_event = 50
			20 = gate_vampire.5
			20 = gate_vampire.6
			20 = gate_vampire.7
			20 = gate_vampire.8
		}
	}

	possible = {
		custom_tooltip = {
			text = je_gate_vampire_emergence_activate
			has_variable = gate_character_vampire
		}
	}

	complete = {
		custom_tooltip = {
			text = je_gate_vampire_emergence_complete
			has_law = law_type:law_gate_vampire_monarchy
		}
	}

	fail = {
		custom_tooltip = {
			text = je_gate_vampire_emergence_fail
			has_variable = gate_rejected_vampirism
		}
	}

	on_complete = {
		hidden_effect = {
			trigger_event = {
				id = gate_vampire.1
				popup = yes
			}
		}
	}

	on_fail = {
		hidden_effect = {
			trigger_event = {
				id = gate_vampire.2
				popup = yes
			}
		}
	}

	progressbar = no

	transferable = no
	can_revolution_inherit = yes

	weight = 200
	should_be_pinned_by_default = yes
}
