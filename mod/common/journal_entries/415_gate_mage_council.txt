io_gate_arcane_council = {
    group = je_group_com_supranational_organizations
    icon = "gfx/interface/icons/generic_icons/gate_magic_innovation.dds"
    weight = 100

    widget = {
        gui = "gui/journal_injects/gate_io_injects.gui"
        name = "gate_inject_mage_council"
        container = "custom_widget_container_je_icon"
    }

    modifiers_while_active = {
        modifier_gate_arcane_council_base
    }

    scripted_button = sb_gate_arcane_council_research_cooperation
    scripted_button = sb_gate_arcane_council_forbid_no_regulations
    scripted_button = sb_gate_arcane_council_forbid_basic_regulations
    scripted_button = sb_gate_arcane_council_cancel_regulations

    immediate = {
        scope:journal_entry = {
            com_set_international_organization_headquaters_effect = {
                state_region = global_var:gate_mage_council_headquarters
            }
            com_hide_all_journal_elements_for_interorg_effect = yes
            random_scope_je_involved = {
                prev = {
                    set_variable = {
                        name = com_interorg_chair_var
                        value = prev.var:gate_magic_researcher
                    }
                }
            }
        }
    }

    on_become_involved_after_activation = {
        set_variable = {
            name = gate_arcane_council_leave_cooldown
            months = 12
        }
        save_scope_as = joining_member
        scope:journal_entry = {
            every_scope_je_involved = {
                limit = {
                    this != root
                }
                post_notification = gate_arcane_council_member_join
            }
        }
    }

    on_no_longer_involved = {
        save_scope_as = leaving_member
        scope:journal_entry = {
            every_scope_je_involved = {
                limit = {
                    this != root
                }
                post_notification = gate_arcane_council_member_leave
            }
        }
    }

    should_be_involved = {
        gate_is_arcane_council_member = yes
        is_gate_magic_country = yes
    }

    should_show_when_not_involved = {
        is_gate_magic_country = yes
    }

    on_monthly_pulse = {
        effect = {
            scope:journal_entry = {
                if = {
                    limit = {
                        has_variable = com_interorg_chair_var
                        OR = {
                            NOT = {
                                exists = var:com_interorg_chair_var
                            }
                            var:com_interorg_chair_var ?= {
                                is_character_alive = no
                            }
                        }
                    }
                    gate_elect_arcane_council_high_mage = yes
                }
            }
        }
    }

    complete = {
        hidden_trigger = {
            always = no
        }
    }

    fail = {
        hidden_trigger = {
            always = no
        }
    }

    should_be_pinned_by_default_involved = yes
    should_be_pinned_by_default_uninvolved_or_context = yes
}
