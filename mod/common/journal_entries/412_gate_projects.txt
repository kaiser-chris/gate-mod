je_gate_project_expand_research_institute = {
	icon = "gfx/interface/icons/event_icons/event_gate_project.dds"

	group = je_group_gate_projects

	widget = {
		gui = "gui/com_journal_injects/injects.gui"
		name = "com_journal_entry_dlc_icon"
		container = "custom_widget_container_7"
	}

	immediate = {
        add_gate_dlc_icon = yes
	}

	complete = {
		gate_project_check_complete = yes
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					capital = {
						has_building = building_gate_magic_research_institute
					}
				}
				gate_project_progress = yes
			}
		}
	}

	invalid = {
		hidden_trigger = {
			OR = {
				has_gate_active_magic_project = no
				AND = {
					has_modifier = modifier_gate_project_institute_new_wing
					has_modifier = modifier_gate_project_institute_improved_instruments
					has_modifier = modifier_gate_project_institute_better_mana_conduits
				}
			}
		}
	}

	on_invalid = {
		hidden_effect = {
			gate_stop_magic_project = yes
		}
	}

	on_complete = {
		if = {
			limit = {
				NOT = {
					has_modifier = modifier_gate_project_institute_new_wing
				}
			}
			add_modifier = modifier_gate_project_institute_new_wing
		}
		else_if = {
			limit = {
				NOT = {
					has_modifier = modifier_gate_project_institute_improved_instruments
				}
			}
			add_modifier = modifier_gate_project_institute_improved_instruments
		}
		else_if = {
			limit = {
				NOT = {
					has_modifier = modifier_gate_project_institute_better_mana_conduits
				}
			}
			add_modifier = modifier_gate_project_institute_better_mana_conduits
		}
		gate_add_magic_innovation = {
			amount = 500
		}
		gate_add_researcher_experience = {
			amount = 10
		}
		hidden_effect = {
			set_variable = gate_has_finished_project
			gate_stop_magic_project = yes
		}
	}

	fail = {
		NOT = {
			capital = {
				has_building = building_gate_magic_research_institute
			}
		}
        has_gate_magic_researcher = no
	}

	on_fail = {
		gate_stop_magic_project = yes
	}

	current_value = {
        value = gate_project_current_progress
	}

	goal_add_value = {
		add = scope:target
	}

    scripted_button = gate_project_button_pause
    scripted_button = gate_project_button_continue
	scripted_button = gate_project_button_stop

	progressbar = yes

	weight = 1000

	should_be_pinned_by_default_uninvolved_or_context = yes
}
