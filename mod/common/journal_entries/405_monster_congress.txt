je_monster_congress = {
	icon = "gfx/interface/icons/event_icons/event_gate_monster.dds"

	group = je_group_magical_affairs

	scripted_button = je_monster_congress_button_start

	widget = {
		gui = "gui/com_journal_injects/injects.gui"
		name = "com_journal_entry_dlc_icon"
		container = "custom_widget_container_7"
	}

    immediate = {
		add_gate_dlc_icon = yes
    }

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_monster_congress_in_progress
				trigger = {
					has_variable = monster_congress_in_progress
				}
			}
			triggered_desc = {
				desc = je_monster_congress_not_started
			}
		}
	}

	is_shown_when_inactive = {
		AND = {
			is_gate_planet_country = no
			OR = {
				has_global_variable = gate_monster_horde_happened
				any_scope_state = {
					state_region = {
						has_harvest_condition = gate_monster_attack
					}
				}
			}
		}
	}

	complete = {
		custom_tooltip = {
			text = je_monster_congress_custom_goal
			scope:journal_entry = {
				scripted_bar_progress = {
					name = pb_monster_congress
					value >= 6 # Months
				}
			}
		}
	}

	invalid = {
		is_gate_planet_country = yes
	}

	on_complete = {
		set_variable = monster_congress_happened
		remove_variable = monster_congress_in_progress
		trigger_event = gate_monster_attack.4
	}

	progressbar = no

	scripted_progress_bar = pb_monster_congress

	transferable = no
	can_revolution_inherit = yes

	weight = 200
	should_be_pinned_by_default_uninvolved_or_context = yes
}
