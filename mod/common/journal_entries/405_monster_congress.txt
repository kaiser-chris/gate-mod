je_gate_monster_congress = {
	icon = "gfx/interface/icons/event_icons/event_gate_monster.dds"

	group = je_group_gate_magical_affairs

	scripted_button = je_gate_monster_congress_button_start

	widget = {
		gui = "gui/journal_injects/gate_monster_congress_injects.gui"
		name = "gate_monster_congress_law_group"
		container = "custom_widget_container_2"
	}

	widget = {
		gui = "gui/com_journal_injects/injects.gui"
		name = "com_journal_entry_dlc_icon"
		container = "custom_widget_container_7"
	}

    immediate = {
		add_gate_dlc_icon = yes
		if = {
			limit = {
				NOT = {
					has_global_variable = gate_monster_horde_happened
				}
			}
			# When it was not a global horde which triggered the JE
			# but a local monster attack
			trigger_event = gate_monster_attack.5
		}
		if = {
			limit = {
				is_ai = yes
			}
			scope:journal_entry = {
				set_variable = monster_congress_in_progress
				if = {
					limit = {
						prev = {
							has_law = law_type:law_traditionalism
						}
					}
					set_variable = {
						name = selected_law
						value = law_type:law_gate_local_monster_hunters
					}
				}
				else_if = {
					limit = {
						prev = {
							ig:ig_industrialists = {
								is_powerful = yes
								is_in_government = yes
							}
							NOT = {
								has_law = law_type:law_traditionalism
							}
							has_technology_researched = currency_standards
						}
					}
					set_variable = {
						name = selected_law
						value = law_type:law_gate_private_monster_hunters
					}
				}
				else_if = {
					limit = {
						prev = {
							ig:ig_armed_forces = {
								is_powerful = yes
								is_in_government = yes
							}
							has_technology_researched = logistics
						}
					}
					set_variable = {
						name = selected_law
						value = law_type:law_gate_militarized_monster_hunters
					}
				}
				else = {
					set_variable = {
						name = selected_law
						value = law_type:law_gate_dedicated_monster_hunters
					}
				}
			}
		}
    }

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_gate_monster_congress_in_progress
				trigger = {
					scope:journal_entry = {
						has_variable = monster_congress_in_progress
					}
				}
			}
			triggered_desc = {
				desc = je_gate_monster_congress_not_started
			}
		}
	}

	is_shown_when_inactive = {
		AND = {
			is_gate_planet_country = no
			OR = {
				has_global_variable = gate_monster_horde_happened
				any_scope_state = {
					state_region = {
						has_harvest_condition = gate_monster_attack
					}
				}
			}
		}
	}

	complete = {
		custom_tooltip = {
			text = je_gate_monster_congress_custom_goal
			scope:journal_entry = {
				scripted_bar_progress = {
					name = pb_gate_monster_congress
					value >= 6 # Months
				}
			}
		}
		custom_tooltip = {
			text = je_gate_monster_congress_button_start_block_no_law
			scope:journal_entry = {
				has_variable = selected_law
			}
		}
	}

	timeout = normal_modifier_time

	invalid = {
		is_gate_planet_country = yes
	}

	on_complete = {
		set_variable = monster_congress_happened
		if = {
			limit = {
				scope:journal_entry = {
					has_variable = selected_law
				}
			}
			activate_law = scope:journal_entry.var:selected_law
		}
		trigger_event = gate_monster_attack.4
	}

	on_timeout = {
		set_variable = monster_congress_happened
		if = {
			limit = {
				prev = {
					has_law = law_type:law_traditionalism
				}
			}
			activate_law = law_type:law_gate_local_monster_hunters
			law_type:law_gate_local_monster_hunters = {
				save_scope_as = selected_law
			}
		}
		else_if = {
			limit = {
				prev = {
					ig:ig_industrialists = {
						is_powerful = yes
						is_in_government = yes
					}
					NOT = {
						has_law = law_type:law_traditionalism
					}
					has_technology_researched = currency_standards
				}
			}
			activate_law = law_type:law_gate_private_monster_hunters
			law_type:law_gate_private_monster_hunters = {
				save_scope_as = selected_law
			}
		}
		else_if = {
			limit = {
				prev = {
					ig:ig_armed_forces = {
						is_powerful = yes
						is_in_government = yes
					}
					has_technology_researched = logistics
				}
			}
			activate_law = law_type:law_gate_militarized_monster_hunters
			law_type:law_gate_militarized_monster_hunters = {
				save_scope_as = selected_law
			}
		}
		else = {
			activate_law = law_type:law_gate_dedicated_monster_hunters
			law_type:law_gate_dedicated_monster_hunters = {
				save_scope_as = selected_law
			}
		}
		trigger_event = gate_monster_attack.4
	}

	progressbar = no

	scripted_progress_bar = pb_gate_monster_congress

	transferable = no
	can_revolution_inherit = yes

	weight = 200
	should_be_pinned_by_default_uninvolved_or_context = yes
}
