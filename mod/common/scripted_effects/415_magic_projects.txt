gate_start_magic_project = {
	custom_tooltip = {
		text = effect_gate_start_magic_project
		set_local_variable = {
			name = gate_project_name
			value = flag:$project$
		}
		set_local_variable = {
			name = monthly_cost
			value = $monthly_cost$
		}
		set_local_variable = {
			name = months
			value = $months$
		}
		add_journal_entry = {
			type = $project$
		}
		set_variable = {
			name = gate_magic_project
			value = je:$project$
		}
	}
}

gate_stop_magic_project = {
	custom_tooltip = {
		text = effect_gate_stop_magic_project
		remove_variable = gate_magic_project
	}
}

gate_project_setup = {
	set_variable = {
		name = gate_magic_project_timer
		value = 0
	}
	set_variable = {
		name = gate_magic_project_length
		value = $time$
	}
	scope:journal_entry = {
		add_modifier = {
			name = gate_modifier_project_cost
			multiplier = $cost$
		}
	}
}

gate_project_progress = {
	if = {
		limit = {
			has_magic_researcher = yes
		}
		change_variable = {
			name = gate_magic_project_timer
			add = 1
		}
	}
}
