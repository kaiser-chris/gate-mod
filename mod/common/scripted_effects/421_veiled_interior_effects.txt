gate_switch_veiled_interior_phase = {
    if = {
        limit = {
            hidden_trigger = {
                flag:je_gate_veiled_interior_phase_diplomacy = flag:$phase$
                exists = je:je_gate_veiled_interior
            }
        }
        custom_tooltip = {
            text = je_gate_veiled_interior_effect_phase_change_diplomacy
            je:je_gate_veiled_interior = {
                set_variable = {
                    name = gate_veiled_interior_phase
                    value = flag:je_gate_veiled_interior_phase_diplomacy
                }
                set_variable = com_hide_status_desc
                com_setup_enactment_journal = {
                    title = je_gate_veiled_interior_diplomacy_title
                    step_progress = 0
                    success_chance = 0.2
                    stall_chance = 0.2
                    current_phase = je_gate_veiled_interior_diplomacy_discussions
                    phase_one = je_gate_veiled_interior_diplomacy_discussions
                    phase_two = je_gate_veiled_interior_diplomacy_alignment
                    phase_three = je_gate_veiled_interior_diplomacy_implementation
                }
                com_set_enactment_journal_icon = {
                    icon = gate_icon_veiled_interior_negotiations
                }
                com_set_enactment_journal_progress_tooltip = {
                    tooltip = je_gate_veiled_interior_diplomacy_tooltip
                }
                gate_veiled_interior_enactment_calculate_chances = yes
            }
        }
    }
    if = {
        limit = {
            hidden_trigger = {
                flag:je_gate_veiled_interior_phase_early_contact = flag:$phase$
                exists = je:je_gate_veiled_interior
            }
        }
        custom_tooltip = {
            text = je_gate_veiled_interior_effect_phase_change_early_contact
            je:je_gate_veiled_interior = {
                set_variable = {
                    name = gate_veiled_interior_phase
                    value = flag:je_gate_veiled_interior_phase_early_contact
                }
                remove_variable = com_hide_status_desc
            }
        }
    }
    if = {
        limit = {
            hidden_trigger = {
                flag:je_gate_veiled_interior_phase_veiled = flag:$phase$
                exists = je:je_gate_veiled_interior
            }
        }
        custom_tooltip = {
            text = je_gate_veiled_interior_effect_phase_change_veiled
            je:je_gate_veiled_interior = {
                set_variable = {
                    name = gate_veiled_interior_phase
                    value = flag:je_gate_veiled_interior_phase_veiled
                }
                remove_variable = com_hide_status_desc
            }
        }
    }
}

gate_advance_veiled_interior_enactment_phase = {
    if = {
        limit = {
            com_is_phase_enactment_journal = {
                phase = je_gate_veiled_interior_diplomacy_discussions
            }
        }
        custom_tooltip = {
            text = je_gate_veiled_interior_effect_phase_change_diplomacy_alignment
            com_set_enactment_journal_current_phase = {
                phase = je_gate_veiled_interior_diplomacy_alignment
            }
        }
    }
    else_if = {
        limit = {
            com_is_phase_enactment_journal = {
                phase = je_gate_veiled_interior_diplomacy_alignment
            }
        }
        custom_tooltip = {
            text = je_gate_veiled_interior_effect_phase_change_diplomacy_implementation
            com_set_enactment_journal_current_phase = {
                phase = je_gate_veiled_interior_diplomacy_implementation
            }
        }
    }
}

gate_veiled_interior_enactment_stall = {
    custom_tooltip = {
        text = je_gate_veiled_interior_effect_diplomacy_stall
        set_local_variable = {
            name = stall_length
            value = $months$
        }
        scope:journal_entry ?= {
            com_set_enactment_journal_stalled = {
                duration = months
                duration_amount = $months$
            }
        }
        fix_variable_error = { variable = stall_length }
    }
}

gate_veiled_interior_enactment_calculate_chances = {
    # Always starts at 15%
    com_set_enactment_journal_stall_chance = {
        value = 0.15
    }

    # Always starts at 5%
    com_set_enactment_journal_success_chance = {
        value = 0.05
    }
}

gate_change_veiled_interior_enactment_stall_chance = {
    custom_tooltip = {
        text = je_gate_veiled_interior_effect_diplomacy_change_stall_change
        scope:journal_entry ?= {
            set_local_variable = {
                name = chance_change
                value = $value$
            }
            set_local_variable = {
                name = chance
                value = {
                    value = com_enactment_journal_stall_chance
                    add = local_var:chance_change
                }
            }
            clamp_local_variable = {
                name = chance
                min = 0
                max = 1
            }
            com_set_enactment_journal_stall_chance = {
                value = local_var:chance
            }
        }
    }
}
