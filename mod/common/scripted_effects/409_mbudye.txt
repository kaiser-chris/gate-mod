add_conversion_progress = {
	hidden_effect = {
		save_temporary_scope_value_as = {
			name = support_change
			value = $support_change$
		}
		save_temporary_scope_value_as = {
			name = conversion_progress
			value = $conversion_progress$
		}
		$country$ = {
			save_temporary_scope_as = affected_country
		}
	}
	if = {
		limit = {
			NOT = {
				$support_change$ = 0
			}
		}
		custom_tooltip = {
			text = je_gate_mbudye_change_support
			je:je_gate_mbudye = {
				add_progress = {
					name = pb_gate_mbudye
					value = $support_change$
				}
			}
		}
	}
	if = {
		limit = {
			NOT = {
				$conversion_progress$ = 0
			}
		}
		custom_tooltip = {
			text = je_gate_mbudye_change_conversion
			je:je_gate_mbudye = {
				if = {
					limit = {
						$country$ = c:LND
					}
					add_progress = {
						name = pb_gate_mbudye_lunda
						value = $conversion_progress$
					}
				}
				else_if = {
					limit = {
						$country$ = c:CHK
					}
					add_progress = {
						name = pb_gate_mbudye_chokwe
						value = $conversion_progress$
					}
				}
				else_if = {
					limit = {
						$country$ = c:KZM
					}
					add_progress = {
						name = pb_gate_mbudye_kazembe
						value = $conversion_progress$
					}
				}
				else_if = {
					limit = {
						$country$ = c:KSN
					}
					add_progress = {
						name = pb_gate_mbudye_kasanje
						value = $conversion_progress$
					}
				}
			}
		}
	}
}

trigger_random_mbudye_investigation_result = {
	random = {
		chance = $chance_success$
		set_variable = mbudye_investigation_success
	}
	if = {
		limit = {
			NOT = {
				has_variable = mbudye_investigation_success
			}
		}
		trigger_event = {
			id = mbudye.6
			days = $investigation_duration$
			popup = yes
		}
	}
	else = {
		trigger_event = {
			id = mbudye.5
			days = $investigation_duration$
			popup = yes
		}
	}
	remove_variable = mbudye_investigation_success
}

form_lunda_nation = {
	set_global_variable = mbudye_unification

	s:STATE_EAST_ANGOLA = {
		add_claim = c:LND
	}
	s:STATE_KATANGA = {
		add_claim = c:LND
	}
	s:STATE_KAZEMBE = {
		add_claim = c:LND
	}
	s:STATE_ZAMBIA = {
		add_claim = c:LND
	}

	c:LND = {
		# Carry over magic nation status
		if = {
			limit = {
				is_magic_country = no
				c:CHK = {
					is_magic_country = yes
				}
			}
			learn_magic = yes
		}
		else_if = {
			limit = {
				is_magic_country = no
				c:KZM = {
					is_magic_country = yes
				}
			}
			learn_magic = yes
		}
		else_if = {
			limit = {
				is_magic_country = no
				c:KSN = {
					is_magic_country = yes
				}
			}
			learn_magic = yes
		}
		annex_with_incorporation = c:CHK
		annex_with_incorporation = c:KZM
		annex_with_incorporation = c:KSN
	}
}

trigger_lunda_uprising = {
	c:LND = {
		# Create basic diplomatic play
		create_diplomatic_play = {
			name = dp_lunda_uprising
			type = dp_independence
			target_country = c:GAD
			war = yes

			add_war_goal = {
				type = annex_country
				holder = c:GAD
				primary_demand = yes
				target_country = c:LND
			}

			add_war_goal = {
				type = return_state
				holder = c:LND
				target_state = s:STATE_ZAMBIA.region_state:GAD
				target_country = c:GAD
				primary_demand = yes
			}
		}

		# Mobilize Armies
		every_military_formation = {
			mobilize_army = yes
		}
	}

	# Add additional war goals depending on circumstances
	every_diplomatic_play = {
		limit = {
			initiator_is = c:LND
			target_is = c:GAD
			is_diplomatic_play_type = dp_independence
		}

		# For later identification
		set_variable = lunda_uprising

		# Add states that could be owned by the Dominion
		if = {
			limit = {
				exists = s:STATE_KATANGA.region_state:GAD
			}
			add_war_goal = {
				type = return_state
				holder = c:LND
				target_state = s:STATE_KATANGA.region_state:GAD
				target_country = c:GAD
				primary_demand = yes
			}
		}
		if = {
			limit = {
				exists = s:STATE_EAST_ANGOLA.region_state:GAD
			}
			add_war_goal = {
				type = return_state
				holder = c:LND
				target_state = s:STATE_EAST_ANGOLA.region_state:GAD
				target_country = c:GAD
				primary_demand = yes
			}
		}
		if = {
			limit = {
				exists = s:STATE_KAZEMBE.region_state:GAD
			}
			add_war_goal = {
				type = return_state
				holder = c:LND
				target_state = s:STATE_KAZEMBE.region_state:GAD
				target_country = c:GAD
				primary_demand = yes
			}
		}

		# Add Botswana to the mix if they do not like
		# the Dominion
		if = {
			limit = {
				c:TSW = {
					this.relations:root < relations_threshold:cordial
				}
			}
			s:STATE_BOTSWANA = {
				add_claim = c:TSW
			}
			s:STATE_NORTHERN_CAPE = {
				add_claim = c:TSW
			}
			add_initiator_backers = {
				c:TSW
			}
			add_war_goal = {
				type = independence
				holder = c:TSW
				target_country = c:GAD
				primary_demand = yes
			}
			if = {
				limit = {
					exists = s:STATE_BOTSWANA.region_state:GAD
				}
				add_war_goal = {
					type = return_state
					holder = c:TSW
					target_state = s:STATE_BOTSWANA.region_state:GAD
					target_country = c:GAD
					primary_demand = yes
				}
			}
			if = {
				limit = {
					exists = s:STATE_NORTHERN_CAPE.region_state:GAD
				}
				add_war_goal = {
					type = return_state
					holder = c:TSW
					target_state = s:STATE_NORTHERN_CAPE.region_state:GAD
					target_country = c:GAD
					primary_demand = yes
				}
			}

			# Mobilize Armies
			c:TSW = {
				every_military_formation = {
					mobilize_army = yes
				}
			}

			# Add reprimanding war goal by GAD
			if = {
				limit = {
					c:TSW = {
						is_subject_type = subject_type_puppet
					}
				}
				add_war_goal = {
					type = annex_country
					holder = c:GAD
					target_country = c:TSW
					primary_demand = yes
				}
			}
			else = {
				add_war_goal = {
					type = reduce_autonomy
					holder = c:GAD
					target_country = c:TSW
					primary_demand = yes
				}
			}
		}
	}
}
