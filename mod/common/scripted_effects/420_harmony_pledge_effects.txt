### Manual Effects
add_gate_economy_harmony_progress = {
    if = {
        limit = {
            exists = je:je_gate_harmony_pledge
        }
        custom_description = {
            text = add_gate_economy_harmony_progress
            subject = this
            value = $amount$

            je:je_gate_harmony_pledge = {
                add_progress = {
                    name = pb_gate_harmony_pledge_economy
                    value = $amount$
                }
                set_gate_harmony_pledge_modifiers = yes
            }
        }
    }
    else = {
        custom_tooltip = gate_create_bug_report
    }
}

### Automatic Effects
# scope = country
gate_hp_population_percentages_cache_update = {
	if = {
		limit = { NOT = { is_country_type = decentralized } }
		save_temporary_scope_as = country
		# Initialize pop counters
		set_variable = {
			name = gate_hp_pop_total
			value = 0
		}
		set_variable = {
			name = gate_hp_pop_citizens
			value = 0
		}
		set_variable = {
			name = gate_hp_pop_gnomish
			value = 0
		}
		set_variable = {
			name = gate_hp_pop_pledge
			value = 0
		}
		# Count the pops
		every_scope_state = {
			if = {
				limit = { is_incorporated = yes }
				every_scope_pop = {
					scope:country = {
						change_variable = {
							name = gate_hp_pop_total
							add = prev.total_size
						}
					}
					if = {
						limit = {
							pop_acceptance >= acceptance_status_4
						}
						scope:country = {
							change_variable = {
								name = gate_hp_pop_citizens
								add = prev.total_size
							}
						}
					}
					if = {
						limit = {
							culture = cu:gnomish
						}
						scope:country = {
							change_variable = {
								name = gate_hp_pop_gnomish
								add = prev.total_size
							}
						}
					}
					if = {
						limit = {
							has_state_religion = yes
						}
						scope:country = {
							change_variable = {
								name = gate_hp_pop_pledge
								add = prev.total_size
							}
						}
					}
				}
			}
		}
		# Save the percents
		scope:country = {
			save_temporary_scope_value_as = {	# Total Pop
				name = gate_hp_tot_population
				value = var:gate_hp_pop_total
			}
			set_variable = {
				name = gate_hp_total_population
				value = scope:gate_hp_tot_population
			}
			save_temporary_scope_value_as = {	# All Citizens
				name = gate_hp_citizens_percent
				value = {
					value = var:gate_hp_pop_citizens
					divide = {
						value = var:gate_hp_pop_total
						min = 1
					}
				}
			}
			set_variable = {
				name = gate_hp_citizenship_percent
				value = scope:gate_hp_citizens_percent
			}
			save_temporary_scope_value_as = {	# Gnomish Portion
				name = gate_hp_gnomish_percent
				value = {
					value = var:gate_hp_pop_gnomish
					divide = {
						value = var:gate_hp_pop_total
						min = 1
					}
				}
			}
			set_variable = {
				name = gate_hp_gnomish_percent
				value = scope:gate_hp_gnomish_percent
			}
			save_temporary_scope_value_as = {	# Pledge Adherents
				name = gate_hp_pledge_percent
				value = {
					value = var:gate_hp_pop_pledge
					divide = {
						value = var:gate_hp_pop_total
						min = 1
					}
				}
			}
			set_variable = {
				name = gate_hp_pledge_percent
				value = scope:gate_hp_pledge_percent
			}
		}
		# Cleanup working variables
		remove_variable = gate_hp_pop_total
		remove_variable = gate_hp_pop_citizens
		remove_variable = gate_hp_pop_gnomish
		remove_variable = gate_hp_pop_pledge
	}
}
