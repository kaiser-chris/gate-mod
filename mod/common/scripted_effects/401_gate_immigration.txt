add_gate_immigration_dwarven = {
	if = {
		limit = {
			has_technology_researched = gate_dwarven_immigration
		}
		add_gate_immigration = {
			culture = dwarven
			modifier = state_gate_monthly_dwarven_immigration_add
		}
	}
}

add_gate_immigration = {
	every_scope_state = {
		limit = {
			has_magic_gate = yes
			modifier:$modifier$ > 0
		}
		if = {
			limit = {
				literacy_rate <= 0.5
			}
			create_pop = {
				culture = $culture$
				size = modifier:$modifier$
				pop_type = officers # Fix illiterate pops
			}
		}
		else_if = {
			limit = {
				literacy_rate > 0.5
				literacy_rate <= 0.6
			}
			create_pop = {
				culture = $culture$
				size = modifier:$modifier$
				pop_type = aristocrats # Fix illiterate pops
			}
		}
		else_if = {
			limit = {
				literacy_rate > 0.6
			}
			create_pop = {
				culture = $culture$
				size = modifier:$modifier$
				pop_type = academics # Fix illiterate pops
			}
		}
	}
}
