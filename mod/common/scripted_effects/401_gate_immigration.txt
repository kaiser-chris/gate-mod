add_gate_immigration_dwarven = {
	if = {
		limit = {
			has_technology_researched = gate_dwarven_immigration
		}
		add_gate_immigration = {
			culture = dwarven
			modifier = state_gate_monthly_dwarven_immigration_add
		}
	}
}

add_gate_immigration = {
	every_scope_state = {
		limit = {
			has_magic_gate = yes
			modifier:$modifier$ > 0
		}
		set_local_variable = {
			name = average_literacy
			value = gate_immiration_average_unemployed_literacy
		}
		set_local_variable = {
			name = average_wealth
			value = gate_immiration_average_unemployed_wealth
		}
		if = {
			limit = {
				local_var:average_wealth > 20
			}
			set_local_variable = {
				name = average_wealth
				value = 20
			}
		}
		if = {
			limit = {
				local_var:average_literacy > 0.6
			}
			set_local_variable = {
				name = average_literacy
				value = 0.6
			}
		}
		create_pop = {
			culture = $culture$
			size = modifier:$modifier$
		}
		every_scope_pop = {
			limit = {
				is_employed = no
			}
			set_pop_literacy = {
				literacy_rate = local_var:average_literacy
			}
			set_pop_wealth = {
				wealth_distribution = local_var:average_wealth
			}
		}
	}
}
