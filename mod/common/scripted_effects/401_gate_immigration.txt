add_gate_immigration_dwarven = {
	if = {
		limit = {
			has_technology_researched = dwarven_gate_immigration
		}
		add_gate_immigration = {
			culture = dwarven
			amount_normal = 10000
			amount_mass = 30000
			modifier = state_gate_monthly_dwarven_immigration_add
		}
	}
}

add_gate_immigration = {
	every_scope_state = {
		limit = {
			has_magic_gate = yes
			modifier:$modifier$ > 0
		}
		# Create pop throws errors after game start
		# but still works
		# Also size can not accept variables or modifiers
		if = {
			limit = {
				modifier:$modifier$ = $amount_normal$
			}
			if = {
				limit = {
					literacy_rate <= 0.5
				}
				create_pop = {
					culture = $culture$
					size = $amount_normal$
					pop_type = officers # Fix illiterate pops
				}
			}
			else_if = {
				limit = {
					literacy_rate > 0.5
					literacy_rate <= 0.6
				}
				create_pop = {
					culture = $culture$
					size = $amount_normal$
					pop_type = aristocrats # Fix illiterate pops
				}
			}
			else_if = {
				limit = {
					literacy_rate > 0.6
				}
				create_pop = {
					culture = $culture$
					size = $amount_normal$
					pop_type = academics # Fix illiterate pops
				}
			}

		}
		else_if = {
			limit = {
				modifier:$modifier$ = $amount_mass$
			}
			if = {
				limit = {
					literacy_rate <= 0.5
				}
				create_pop = {
					culture = $culture$
					size = $amount_mass$
					pop_type = officers # Fix illiterate pops
				}
			}
			else_if = {
				limit = {
					literacy_rate > 0.5
					literacy_rate <= 0.6
				}
				create_pop = {
					culture = $culture$
					size = $amount_mass$
					pop_type = aristocrats # Fix illiterate pops
				}
			}
			else_if = {
				limit = {
					literacy_rate > 0.6
				}
				create_pop = {
					culture = $culture$
					size = $amount_mass$
					pop_type = academics # Fix illiterate pops
				}
			}
		}
	}
}
