gate_ai_conditional_research = {
    if = {
        limit = {
            NOT = {
                has_technology_researched = $research$
            }
            $research$ = yes
            var:magic_innovation >= gate_ai_research_cost
        }
        add_technology_researched = $research$
        change_variable = {
            name = magic_innovation
            subtract = gate_ai_research_cost
        }
    }
}
