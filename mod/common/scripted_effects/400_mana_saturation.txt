gate_calculate_mana_saturation = {
	set_global_variable = {
		name = global_mana_saturation
		value = {
			value = 0
			every_state_region = {
				limit = {
                    is_state_region_land = yes
					var:mana_density > 0
				}
				add = var:mana_density
			}
			divide = global_var:state_count
		}
	}
}

gate_propogate_mana_density = {
	every_state_region = {
		limit = {
			has_variable = mana_density
			var:mana_density >= gate_medium_mana_density_threshhold
            is_state_region_land = yes
		}
		save_temporary_scope_value_as = {
			name = origin_density
			value = var:mana_density
		}
		every_neighbouring_state = {
			limit = {
				state_region = {
					var:mana_density < gate_maximum_mana_density
					var:mana_density < scope:origin_density
				}
			}
			state_region = {
				gate_add_mana_density = {
					amount = $amount$
				}
			}
		}
		if = {
			limit = {
				has_variable_list = mana_propogation_states
			}
			every_in_list = {
				variable = mana_propogation_states
				limit = {
					var:mana_density < gate_maximum_mana_density
					var:mana_density < scope:origin_density
				}
				gate_add_mana_density = {
					amount = $amount$
				}
			}
		}
	}
}

gate_add_mana_density = {
	if = {
		limit = {
			has_variable = mana_density
			var:mana_density < gate_maximum_mana_density
		}
		save_temporary_scope_value_as = {
			name = added_mana_density
			value = $amount$
		}
		custom_tooltip = {
			text = effect_gate_add_mana_density
			# Propagation
			change_variable = {
				name = mana_density
				add = $amount$
			}
			clamp_variable = {
				name = mana_density
				max = gate_maximum_mana_density
				min = gate_minimum_mana_density
			}
			gate_add_mana_density_modifier = yes
		}
	}
}

gate_add_mana_density_modifier = {
	every_scope_state = {
		if = {
			limit = {
				state_region = {
					var:mana_density >= gate_medium_mana_density_threshhold
					var:mana_density < gate_high_mana_density_threshhold
				}
				NOT = {
					has_modifier = medium_mana_density_modifier
				}
			}
			if = {
				limit = {
					has_modifier = high_mana_density_modifier
				}
				remove_modifier = high_mana_density_modifier
			}
			add_modifier = medium_mana_density_modifier
		}
		else_if = {
			limit = {
				state_region = {
					var:mana_density >= gate_high_mana_density_threshhold
				}
				NOT = {
					has_modifier = high_mana_density_modifier
				}
			}
			if = {
				limit = {
					has_modifier = medium_mana_density_modifier
				}
				remove_modifier = medium_mana_density_modifier
			}
			add_modifier = high_mana_density_modifier
		}
	}
}

initial_gate_propogate_mana_density = {
	# Every state with a gate starts with
	# 100% mana density
	every_state_region = {
		limit = {
			has_magic_gate = yes
            is_state_region_land = yes
		}
		set_variable = {
			name = mana_density
			value = gate_maximum_mana_density
		}
		gate_add_mana_density_modifier = yes
	}

	# Hidden Gate in Amazonas
	s:STATE_AMAZONAS = {
		set_variable = gate_veiled_gate
		set_variable = {
			name = mana_density
			value = gate_maximum_mana_density
		}
		gate_add_mana_density_modifier = yes
	}

	# Propagate initial mana density fall off
	every_state_region = {
		limit = {
            trigger_if = {
                limit = {
                    is_state_region_land = yes
                }
                var:mana_density = 100
            }
            trigger_else = {
                always = no
            }
		}
		gate_propogate_from_state_region = {
			amount = 70
		}
	}
	every_state_region = {
		limit = {
            trigger_if = {
                limit = {
                    is_state_region_land = yes
                }
                var:mana_density = 70
            }
            trigger_else = {
                always = no
            }
		}
		gate_propogate_from_state_region = {
			amount = 40
		}
	}
	every_state_region = {
		limit = {
            trigger_if = {
                limit = {
                    is_state_region_land = yes
                }
                var:mana_density = 40
            }
            trigger_else = {
                always = no
            }
		}
		gate_propogate_from_state_region = {
			amount = 10
		}
	}
}

gate_propogate_from_state_region = {
	save_temporary_scope_value_as = {
		name = origin_density
		value = var:mana_density
	}
	every_neighbouring_state = {
		limit = {
			state_region = {
				var:mana_density < gate_maximum_mana_density
				var:mana_density < scope:origin_density
			}
		}
		state_region = {
			set_variable = {
				name = mana_density
				value = $amount$
			}
			gate_add_mana_density_modifier = yes
		}
	}
	if = {
		limit = {
			has_variable_list = mana_propogation_states
		}
		every_in_list = {
			variable = mana_propogation_states
			limit = {
				var:mana_density < gate_maximum_mana_density
				var:mana_density < scope:origin_density
			}
			set_variable = {
				name = mana_density
				value = $amount$
			}
			gate_add_mana_density_modifier = yes
		}
	}
}

gate_setup_mana_density = {
    # Get amount of states for
    # later calculation and
    # initialize mana density variable
    set_global_variable = {
        name = state_count
        value = 0
    }
    every_state_region = {
        limit = {
            is_state_region_land = yes
        }
        change_global_variable = {
            name = state_count
            add = 1
        }
        set_variable = {
            name = mana_density
            value = 0
        }
    }

    set_global_variable = {
        name = global_mana_saturation
        value = 0
    }

    # Build initial Mana Density around gates
    initial_gate_propogate_mana_density = yes

    # Randomly propagate mana density
    # so every start is a little different
    set_local_variable = {
        name = gate_mana_propagation_amount
        value = { 1 4 }
    }
    gate_propogate_mana_density = {
        amount = local_var:gate_mana_propagation_amount
    }

    # Calculate initial saturation
    gate_calculate_mana_saturation = yes
}
