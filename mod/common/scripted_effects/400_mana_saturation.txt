calculate_mana_saturation = {
	set_global_variable = {
		name = global_mana_saturation
		value = {
			value = global_var:global_mana_density_cache
			divide = global_var:state_count
		}
	}
}

gate_propogate_mana_density = {
	every_state_region = {
		limit = {
			has_variable = mana_density
			var:mana_density >= medium_mana_density_threshhold
		}
		save_temporary_scope_value_as = {
			name = origin_density
			value = var:mana_density
		}
		every_neighbouring_state = {
			limit = {
				state_region = {
					var:mana_density < maximum_mana_density
					var:mana_density < scope:origin_density
				}
			}
			state_region = {
				add_mana_density = {
					amount = 1
				}
			}
		}
		if = {
			limit = {
				has_variable_list = mana_propogation_states
			}
			every_in_list = {
				variable = mana_propogation_states
				limit = {
					var:mana_density < maximum_mana_density
					var:mana_density < scope:origin_density
				}
				add_mana_density = {
					amount = 1
				}
			}
		}
	}
}

add_mana_density = {
	if = {
		limit = {
			has_variable = mana_density
		}
		# Propagation
		change_variable = {
			name = mana_density
			add = $amount$
		}
		clamp_variable = {
			name = mana_density
			max = maximum_mana_density
			min = minimum_mana_density
		}
		# Cache update
		change_global_variable = {
			name = global_mana_density_cache
			add = {
				value = $amount$
				add = 0.45 # Add 45% so it is guaranteed to hit it
			}
		}
		add_mana_density_modifier = yes
	}
}

add_mana_density_modifier = {
	every_scope_state = {
		if = {
			limit = {
				state_region = {
					var:mana_density >= medium_mana_density_threshhold
					var:mana_density < high_mana_density_threshhold
				}
				NOT = {
					has_modifier = medium_mana_density_modifier
				}
			}
			if = {
				limit = {
					has_modifier = high_mana_density_modifier
				}
				remove_modifier = high_mana_density_modifier
			}
			add_modifier = medium_mana_density_modifier
		}
		else_if = {
			limit = {
				state_region = {
					var:mana_density >= high_mana_density_threshhold
				}
				NOT = {
					has_modifier = high_mana_density_modifier
				}
			}
			if = {
				limit = {
					has_modifier = medium_mana_density_modifier
				}
				remove_modifier = medium_mana_density_modifier
			}
			add_modifier = high_mana_density_modifier
		}
	}
}

initial_gate_propogate_mana_density = {
	# Every state with a gate starts with
	# 100% mana density
	every_state_region = {
		limit = {
			has_magic_gate = yes
		}
		set_variable = {
			name = mana_density
			value = maximum_mana_density
		}
		add_mana_density_modifier = yes
	}

	# Speed up propagation in America by
	# setting some states surrounding the
	# gate to 100% mana density as well
	s:STATE_GUATEMALA = {
		set_variable = {
			name = mana_density
			value = maximum_mana_density
		}
		add_mana_density_modifier = yes
	}
	s:STATE_HONDURAS = {
		set_variable = {
			name = mana_density
			value = maximum_mana_density
		}
		add_mana_density_modifier = yes
	}
	s:STATE_NICARAGUA = {
		set_variable = {
			name = mana_density
			value = maximum_mana_density
		}
		add_mana_density_modifier = yes
	}
	s:STATE_COSTA_RICA = {
		set_variable = {
			name = mana_density
			value = maximum_mana_density
		}
		add_mana_density_modifier = yes
	}
	s:STATE_PANAMA = {
		set_variable = {
			name = mana_density
			value = maximum_mana_density
		}
		add_mana_density_modifier = yes
	}
	s:STATE_ANTIOQUIA = {
		set_variable = {
			name = mana_density
			value = maximum_mana_density
		}
		add_mana_density_modifier = yes
	}
	s:STATE_CHIAPAS = {
		set_variable = {
			name = mana_density
			value = maximum_mana_density
		}
		add_mana_density_modifier = yes
	}

	# Propagate initial mana density fall off
	every_state_region = {
		limit = {
			var:mana_density = 100
		}
		propogate_from_state_region = {
			amount = 70
		}
	}
	every_state_region = {
		limit = {
			var:mana_density = 70
		}
		propogate_from_state_region = {
			amount = 40
		}
	}
	every_state_region = {
		limit = {
			var:mana_density = 40
		}
		propogate_from_state_region = {
			amount = 10
		}
	}
}

propogate_from_state_region = {
	save_temporary_scope_value_as = {
		name = origin_density
		value = var:mana_density
	}
	every_neighbouring_state = {
		limit = {
			state_region = {
				var:mana_density < maximum_mana_density
				var:mana_density < scope:origin_density
			}
		}
		state_region = {
			set_variable = {
				name = mana_density
				value = $amount$
			}
			add_mana_density_modifier = yes
		}
	}
	if = {
		limit = {
			has_variable_list = mana_propogation_states
		}
		every_in_list = {
			variable = mana_propogation_states
			limit = {
				var:mana_density < maximum_mana_density
				var:mana_density < scope:origin_density
			}
			set_variable = {
				name = mana_density
				value = $amount$
			}
			add_mana_density_modifier = yes
		}
	}
}
