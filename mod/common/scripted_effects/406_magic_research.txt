gate_calculate_magic_research = {
    # Banking
    if = {
        limit = {
            has_variable = magic_innovation
        }
        change_variable = {
            name = magic_innovation
            add = gate_magic_innovation
        }
    }
    else = {
        set_variable = {
            name = magic_innovation
            value = gate_magic_innovation
        }
    }

    clamp_variable = {
        name = magic_innovation
        min = 0
        max = global_var:magic_innovation_bank_max
    }
}

# Research Institutes can only exist in capital states
gate_cleanup_research_institutes = {
	every_scope_state = {
		limit = {
			is_capital = no
			has_building = building_gate_magic_research_institute
		}
		remove_building = building_gate_magic_research_institute
	}
}

gate_reset_research_institute_pm = {
	if = {
		limit = {
			capital = {
				has_building = building_gate_magic_research_institute
			}
		}
		if = {
			limit = {
				NOT = {
					has_variable = gate_magic_researcher
				}
				capital = {
					any_scope_building = {
						is_building_type = building_gate_magic_research_institute
						NOT = {
							has_active_production_method = pm_gate_no_researcher
						}
					}
				}
			}
			activate_production_method = {
				building_type = building_gate_magic_research_institute
				production_method = pm_gate_no_researcher
			}
		}
		else_if = {
			limit = {
				var:gate_magic_researcher ?= {
					has_variable = magic_experience
					var:magic_experience >= gate_character_master_threshhold
				}
				capital = {
					any_scope_building = {
						is_building_type = building_gate_magic_research_institute
						NOT = {
							has_active_production_method = pm_gate_archmage_researcher
						}
					}
				}
			}
			activate_production_method = {
				building_type = building_gate_magic_research_institute
				production_method = pm_gate_archmage_researcher
			}
		}
		else_if = {
			limit = {
				var:gate_magic_researcher ?= {
					has_variable = magic_experience
					var:magic_experience >= gate_character_experienced_threshhold
				}
				capital = {
					any_scope_building = {
						is_building_type = building_gate_magic_research_institute
						NOT = {
							has_active_production_method = pm_gate_experienced_researcher
						}
					}
				}
			}
			activate_production_method = {
				building_type = building_gate_magic_research_institute
				production_method = pm_gate_experienced_researcher
			}
		}
		else_if = {
			limit = {
				capital = {
					any_scope_building = {
						is_building_type = building_gate_magic_research_institute
						NOT = {
							has_active_production_method = pm_gate_novice_researcher
						}
					}
				}
			}
			activate_production_method = {
				building_type = building_gate_magic_research_institute
				production_method = pm_gate_novice_researcher
			}
		}
	}
}

gate_make_magic_researcher_busy = {
	var:gate_magic_researcher ?= {
		set_character_busy_and_immortal = yes
		set_variable = {
			name = com_opinion
			value = flag:character_opinion_magic_researcher_busy
		}
	}
}

gate_make_magic_researcher_not_busy = {
	var:gate_magic_researcher ?= {
		set_character_busy_and_immortal = no
		set_variable = {
			name = com_opinion
			value = flag:character_opinion_magic_researcher
		}
	}
}

gate_cleanup_researchers = {
	if = {
		limit = {
			has_variable = gate_magic_researcher
			OR = {
				NOT = {
					exists = var:gate_magic_researcher
				}
				var:gate_magic_researcher ?= {
					is_character_alive = no
				}
			}
		}
		if = {
			limit = {
				has_journal_entry = je_gate_researching_magic
			}
			je:je_gate_researching_magic = {
				clear_variable_list = com_journal_characters
			}
		}
		activate_production_method = {
			building_type = building_gate_magic_research_institute
			production_method = pm_gate_no_researcher
		}
		remove_variable = gate_magic_researcher
	}
}

gate_add_magic_innovation = {
	set_local_variable = {
		name = added_innovation
		value = $amount$
	}
	custom_tooltip = {
		text = effect_gate_add_magic_innovation
		change_variable = {
			name = magic_innovation
			add = local_var:added_innovation
		}
		clamp_variable = {
			name = magic_innovation
			min = 0
			max = global_var:magic_innovation_bank_max
		}
	}
}

gate_hire_researcher = {
	trigger_event = {
		id = researching_magic.4
		weeks = 1
	}
	set_variable = {
		name = actively_hiring_magic_researcher
		weeks = 1
	}
}

gate_create_and_activate_magic_researcher = {
	if = {
		limit = {
			is_gate_magic_country = yes
		}
		create_character = {
			template = gate_magic_researcher_template
			interest_group = ig_gate_magic_users
			save_scope_as = magic_researcher
		}
	}
	else = {
		create_character = {
			template = gate_magic_researcher_template
			interest_group = ig_intelligentsia
			save_scope_as = magic_researcher
		}
	}
	scope:magic_researcher = {
		gate_activate_magic_researcher = yes
	}
	post_notification = message_gate_arcanist_hired
}

gate_activate_magic_researcher = {
	custom_tooltip = {
		text = effect_gate_activate_magic_researcher
		set_local_variable = {
			name = gate_magic_researcher
			value = this
		}

		set_variable = is_magic_researcher

		if = {
			limit = {
				NOT = {
					has_variable = com_opinion
				}
			}
			set_variable = {
				name = com_opinion
				value = flag:character_opinion_magic_researcher
			}
		}

		set_local_variable = {
			name = added_experience
			value = {
				integer_range = {
					min = 5
					max = 15
				}
			}
		}
		gate_increase_character_experience = {
			amount = local_var:added_experience
		}

		replace_character_roles = {
			remove_all = yes
			add = {
				gate_active_arcanist
			}
		}

		save_scope_as = magic_researcher
		owner = {
			post_notification = message_gate_arcanist_hired
			if = {
				limit = {
					var:gate_magic_researcher ?= {
						is_character_alive = yes
					}
				}
				var:gate_magic_researcher ?= {
					retire_character = {
						value = yes
						hidden = yes
					}
				}
			}
			set_variable = {
				name = gate_magic_researcher
				value = prev
			}
			gate_reset_research_institute_pm = yes
			if = {
				limit = {
					has_journal_entry = je_gate_researching_magic
				}
				je:je_gate_researching_magic = {
					clear_variable_list = com_journal_characters
					add_to_variable_list = {
						name = com_journal_characters
						target = prev.var:gate_magic_researcher
					}
				}
				# Our researcher complaining about the missing institute
				if = {
					limit = {
						capital = {
							NOT = {
								has_building = building_gate_magic_research_institute
							}
						}
					}
					var:gate_magic_researcher = {
						set_variable = {
							name = com_opinion
							value = flag:character_opinion_magic_researcher_missing_institute
						}
					}
				}
			}
		}
	}

}
