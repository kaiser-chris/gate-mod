gate_cycle_tabs = {
    if = {
        # When there are more tabs cycle to next one
        limit = {
            has_variable = $tab_variable$
            var:$tab_variable$ < $tab_count$
        }
        change_variable = {
            name = $tab_variable$
            add = 1
        }
    }
    else = {
        # If there is no variable we set it
        set_variable = {
            name = $tab_variable$
            value = 1
        }
    }
}

add_gate_magic_main_button = {
    if = {
        limit = {
            NOT = {
                is_target_in_global_variable_list = {
                    name = custom_button_list_flag
                    target = flag:gate_button_main
                }
            }
        }
        add_to_global_variable_list = {
            name = custom_button_list_flag
            target = flag:gate_button_main
        }
    }
}

gate_order_list_with_reverse = {
    if = {
        limit = {
            scope:reverse = yes
        }
        sort_variable_list = {
            name = $list$
            order = {
                value = $order$
                multiply = -1
            }
        }
        error_log = "reverse"
    }
    else = {
        sort_variable_list = {
            name = $list$
            order = $order$
        }
        error_log = "normal"
    }
}
