building_elven_forest = {
	building_group = bg_elven_forests
	icon = "gfx/interface/icons/building_icons/building_elven_forest.dds"
	city_type = none
	levels_per_mesh = 5

	can_build_private = {
		always = no
	}

	has_max_level = yes
	downsizeable = no
	terrain_manipulator = forestry

	ai_value = 100

	unlocking_technologies = {
		elven_forests
	}
	ownership_type = no_ownership

	potential = {
		owner = {
			has_technology_researched = elven_forests
		}
		# Are there even fields/forests
		arable_land > 0
		# Only one elven forest per state region
		# to prevent split state abuse
		state_region = {
			NOT = {
				has_building = building_elven_forest
			}
		}
	}

	possible = {
		custom_tooltip = {
			text = building_requirement_medium_mana_density
			state_region = {
				var:mana_density >= medium_mana_density_threshhold
			}
		}
	}

	production_method_groups = {
		pmg_elven_forest
	}

	required_construction = construction_cost_medium

	background = "gfx/interface/icons/building_icons/backgrounds/building_panel_bg_magic.dds"
}
