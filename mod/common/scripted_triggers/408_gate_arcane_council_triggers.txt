gate_is_arcane_council_leader = {
    custom_description = {
        text = gate_is_arcane_council_leader
        subject = this

        je:io_gate_arcane_council.var:com_interorg_chair_var ?= {
            owner = prev
        }
    }
}

gate_is_arcane_council_member = {
    custom_description = {
        text = gate_is_arcane_council_member
        subject = this

        OR = {
            has_variable = gate_joined_mage_council
            je:io_gate_arcane_council ?= {
                any_scope_je_involved = {
                    this = root
                }
            }
        }
    }
}

gate_recently_joined_arcane_council = {
    custom_description = {
        text = gate_recently_joined_arcane_council
        subject = this

        has_variable = gate_arcane_council_leave_cooldown
    }
}

gate_trigger_incident_events = {
    is_magic_country = yes
    NOT = {
        has_law_or_variant = law_type:law_closed_borders
    }
    any_country = {
        NOT = {
            this = root
        }
        is_magic_country = yes
        has_overlapping_interests = root
    }
    global_variable_list_size = {
        name = magic_countries
        value > 10
    }
    NOT = {
        has_variable = gate_foreign_mage_incident_cooldown
    }
}
