# Check whether a specific interaction
# is valid for the given expedition leader
gate_expedition_interaction_valid = {
	OR = {
		# Always available
		scope:interaction.var:type ?= gate_expedition_interaction_home
		scope:interaction.var:type ?= gate_expedition_interaction_explore
		AND = {
			# Militarist
			var:gate_expedition.var:leader_type ?= gate_expedition_leader_militarist
			OR = {
				scope:interaction.var:type ?= gate_expedition_interaction_raid
				scope:interaction.var:type ?= gate_expedition_interaction_conquer
			}
		}
		AND = {
			# Politician
			var:gate_expedition.var:leader_type ?= gate_expedition_leader_politician
			scope:interaction.var:type ?= gate_expedition_interaction_contact
		}
		AND = {
			# Industrialist
			var:gate_expedition.var:leader_type ?= gate_expedition_leader_industrialist
			scope:interaction.var:type ?= gate_expedition_interaction_trade
		}

	}
}

gate_expedition_has_budget = {
	save_temporary_scope_value_as = {
		name = budget_cost
		value = $amount$
	}
	custom_tooltip = {
		text = trigger_gate_expedition_budget
		var:gate_expedition ?= {
			trigger_if = {
				limit = {
					exists = var:expedition_budget
				}
				var:expedition_budget >= $amount$
			}
			trigger_else = {
				always = no
			}
		}
	}
}

gate_expedition_has_covered = {
	custom_tooltip = {
		text = trigger_gate_expedition_has_covered
		var:gate_expedition ?= {
			var:location ?= {
				any_in_list = {
					variable = states
					has_variable = hidden
				}
			}
		}
	}
}

gate_expedition_has_no_activity = {
	custom_tooltip = {
		text = trigger_gate_expedition_has_no_activity
		var:gate_expedition ?= {
			NOT = {
				has_variable = activity
			}
		}
	}
}
