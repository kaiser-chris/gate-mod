gnomish_court_portrait_trigger = {
    scope:character ?= {
        NOT = {
            AND = {
                exists = home_country
                is_ruler_of_any_country = yes
                home_country = {
                    OR = {
                        has_law = law_type:law_monarchy
                        has_law = law_type:law_theocracy
                    }
                }
            }
        }
        has_culture = cu:gnomish
    }
}
gnomish_portrait_pop_trigger = {
    scope:pop ?= {
        culture = cu:gnomish
    }
}
