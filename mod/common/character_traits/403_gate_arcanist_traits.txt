gate_arcanist_magic_sight = {
    type = skill

    texture = gfx/interface/icons/character_trait_icons/gate_arcanist_magic_sight.dds

    character_modifier = {
        character_popularity_add = -15
        character_prominence_add = 10
    }

    possible = {
        OR = {
            has_role = gate_arcanist
            has_role = gate_active_arcanist
            owner = {
                is_gate_magic_country = yes
            }
        }
    }

    weight = {
        value = 2

        # More likely for experienced characters
        if = {
            limit = {
                has_variable = magic_experience
            }
            multiply = {
                value = var:magic_experience
                divide = 2
            }
        }

        # More likely for arcanists
        if = {
            limit = {
                OR = {
                    has_role = gate_arcanist
                    has_role = gate_active_arcanist
                }
            }
            add = 10
        }
    }

    value = 2
}
